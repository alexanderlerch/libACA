82int CKey::compKey()
83{
84 if (!m_bIsInitialized)
85 return -1;
86
87 int aiDim[2] = { 0 };
88 float **m_ppfPitchChromagram = 0;
89 float afAvgPitchChroma[12] = { 0 };
90
91 // alloc memory
92 m_pCPitchChromaExtractor->getFeatureDimensions(aiDim[0], aiDim[1]);
93 CMatrix::alloc(m_ppfPitchChromagram, aiDim[0], aiDim[1]);
94
95 // compute pitch chromagram
96 m_pCPitchChromaExtractor->compFeatureNDim(m_ppfPitchChromagram);
97
98 // average pitch chroma
99 for (auto n = 0; n < aiDim[0]; n++)
100 afAvgPitchChroma[n] = CVector::getMean(m_ppfPitchChromagram[n], aiDim[1]);
101
102 CMatrix::free(m_ppfPitchChromagram, aiDim[0]);
103
104 // compute key
105 return m_pCKeyFromChroma->getKey(afAvgPitchChroma);
106}
