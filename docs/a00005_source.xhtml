221Error_t CAudioFileIf::getPosition( double &dTimeInS )
222{
223 long long iFrame;
224
225 // check file status
226 if (!isOpen())
227 return Error_t::kUnknownError;
228
229 // check file properties
230 if (!isInitialized())
231 return Error_t::kNotInitializedError;
232
233 dTimeInS = -1.;
234 Error_t iErr = getPosition(iFrame);
235
236 if (iErr != Error_t::kNoError)
237 return iErr;
238
239 dTimeInS = iFrame * (1./m_CurrFileSpec.fSampleRateInHz);
240 return Error_t::kNoError;
