196Error_t CChordIf::compChords(Chords_t* peChord, bool bWithViterbi /*= true*/)
197{
198 if (!m_bIsInitialized)
199 return Error_t::kFunctionIllegalCallError;
200 if (!peChord)
201 return Error_t::kFunctionInvalidArgsError;
202
203 assert(m_pfProcessBuff1);
204 assert(m_pCBlockAudio);
205 assert(m_pCNormalize);
206 assert(m_pCChord);
207
208 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
209
210 for (auto n = 0; n < iNumBlocks; n++)
211 {
212 float afChordProb[kNumChords] = { 0 };
213
214 // retrieve the next audio block
215 m_pCBlockAudio->getNextBlock(m_pfProcessBuff1);
216
217 // normalize if specified
218 if (m_pCNormalize)
219 m_pCNormalize->normalizeBlock(m_pfProcessBuff1, m_iBlockLength);
220
221 assert(m_pfProcessBuff2);
222 assert(m_pCFft);
223 computeMagSpectrum_();
224
225 // compute instantaneous chord probs
226 m_pCChord->compChordProb(afChordProb, m_pfProcessBuff1);
227
228 // store result
229 if (bWithViterbi)
230 CMatrix::setCol(m_ppfChordProbs, afChordProb, n, kNumChords);
231 else
232 {
233 float fTmp = 0;
234 long long iMaxIdx = -1;
235 CVector::findMax(afChordProb, fTmp, iMaxIdx, kNumChords);
236 peChord[n] = static_cast<Chords_t>(iMaxIdx);
237 }
238 }
239
240 // compute viterbi
241 if (bWithViterbi)
242 {
243 int* piTmp = 0;
244 CVector::alloc(piTmp, iNumBlocks);
245 m_pCViterbi->compViterbi(m_ppfChordProbs);
246 m_pCViterbi->getStateSequence(piTmp);
247
248 for (auto n = 0; n < iNumBlocks; n++)
249 peChord[n] = static_cast<Chords_t>(piTmp[n]);
250
251 CVector::free(piTmp);
252 }
253
254 return Error_t::kNoError;
255}

413void CChordIf::initViterbi_()
414{
415 int iNumBlocks = static_cast<int>(m_pCBlockAudio->getNumBlocks());
416
417 /////////////////////////////
418 // start probabilities
419 float afStartProb[kNumChords] = { 0 };
420
421 CVector::setValue(afStartProb, 1.F / (kNumChords + 1), kNumChords);
422 afStartProb[kNoChord] *= 2.F;
423
424 /////////////////////////////
425 // transition probabilities
426 float** ppfTransProb = 0;
427 CMatrix::alloc(ppfTransProb, kNumChords, kNumChords);
428 const int aiCircleOfFifths[24] = { 0, -5, 2, -3, 4, -1, 6, 1, -4, 3, -2, 5, -3, 4, -1, 6, 1, -4, 3, -2, 5, 0, -5, 2 };
429 float fRadius = 1.F;
430 float fDistance = .5F;
431
432 for (auto m = 0; m < kNumChords-1; m++)
433 {
434 float fX1 = static_cast<float>(fRadius * std::cos(2. * M_PI * aiCircleOfFifths[m] / 12.));
435 float fY1 = static_cast<float>(fRadius * std::sin(2. * M_PI * aiCircleOfFifths[m] / 12.));
436 float fZ1 = fDistance - ((m < 12) ? 0 : fDistance);
437 for (auto n = 0; n < kNumChords - 1; n++)
438 {
439 float fXDist = static_cast<float>(fX1 - fRadius * std::cos(2. * M_PI * aiCircleOfFifths[n] / 12.));
440 float fYDist = static_cast<float>(fY1 - fRadius * std::sin(2. * M_PI * aiCircleOfFifths[n] / 12.));
441 float fZDist = fZ1 - (fDistance - ((n < 12) ? 0 : fDistance));
442
443 ppfTransProb[m][n] = .1F + std::sqrt(fXDist * fXDist + fYDist * fYDist + fZDist * fZDist);
444 }
445 }
446 // convert to probabilities
447 CMatrix::mulC_I(ppfTransProb, -1.F / (.1F + CMatrix::getMax(ppfTransProb, kNumChords, kNumChords)), kNumChords, kNumChords);
448 CMatrix::addC_I(ppfTransProb, 1.F, kNumChords, kNumChords);
449
450 // no chord probs
451 CVector::setValue(ppfTransProb[kNoChord], 1.F / kNumChords, kNumChords);
452 CMatrix::setCol(ppfTransProb, ppfTransProb[kNoChord], kNoChord, kNumChords);
453
454 // normalization
455 for (auto m = 0; m < kNumChords; m++)
456 CVector::mulC_I(ppfTransProb[m], 1.F / CVector::getSum(ppfTransProb[m], kNumChords), kNumChords);
457
458 /////////////////////////////
459 // instance creation and initialization
460 m_pCViterbi = new CViterbi();
461 m_pCViterbi->init(ppfTransProb, afStartProb, kNumChords, iNumBlocks);
462
463 // free temp memory
464 CMatrix::free(ppfTransProb, kNumChords);
465}
