193Error_t CPitchIf::compF0(float* pfPitch)
194{
195 if (!m_bIsInitialized)
196 return Error_t::kFunctionIllegalCallError;
197 if (!pfPitch)
198 return Error_t::kFunctionInvalidArgsError;
199
200 assert(m_pfProcessBuff1);
201 assert(m_pCBlockAudio);
202 assert(m_pCNormalize);
203 assert(m_pCPitch);
204
205 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
206
207 for (auto n = 0; n < iNumBlocks; n++)
208 {
209 // retrieve the next audio block
210 m_pCBlockAudio->getNextBlock(m_pfProcessBuff1);
211
212 // normalize if specified
213 if (m_pCNormalize)
214 m_pCNormalize->normalizeBlock(m_pfProcessBuff1, m_iBlockLength);
215
216 if (isPitchExtractorSpectral_(m_pCPitch->getPitchExtractorIdx()))
217 {
218 assert(m_pfProcessBuff2);
219 assert(m_pCFft);
220 computeMagSpectrum_();
221 }
222
223 pfPitch[n] = m_pCPitch->compF0(m_pfProcessBuff1);
224 }
225
226 return Error_t::kNoError;
227}

300Error_t CPitchIf::init_(PitchExtractors_t ePitchIdx)
301{

304 // initialize FFT and fft buffer
305 m_pCFft = new CFft();
306 m_pCFft->init(m_iBlockLength);
307 // allocate processing memory

309 CVector::alloc(m_pfProcessBuff2, m_pCFft->getLength(CFft::kLengthFft));
310 CPitchFromBlockIf::create(m_pCPitch, ePitchIdx, m_pCFft->getLength(CFft::kLengthMagnitude), m_fSampleRate);

314 // allocate processing memory

316 CPitchFromBlockIf::create(m_pCPitch, ePitchIdx, m_iBlockLength, m_fSampleRate);
317 }
318 m_bIsInitialized = true;
319
320 return Error_t::kNoError;
