24Error_t CFft::init( int iBlockLength, int iZeroPadFactor, WindowFunction_t eWindow /*= kWindowHann*/, Windowing_t eWindowing /*= kPreWindow*/ )
25{
26 Error_t rErr = Error_t::kNoError;
27
28 // sanity check
29 if (iZeroPadFactor <= 0)
30 return Error_t::kFunctionInvalidArgsError;
31
32 // clean up
33 reset();
34
35 // make sure the fft length is a power of two
36 int iInternalBlockLength = CUtil::isPowOf2(iBlockLength) ? iBlockLength : CUtil::nextPowOf2(iBlockLength);
37 m_iDataLength = iBlockLength;
38 m_iFftLength = iInternalBlockLength * iZeroPadFactor;
39
40 m_ePrePostWindowOpt = eWindowing;
41
42 rErr = allocMemory_ ();
43 if (rErr != Error_t::kNoError)
44 return rErr;
45
46 rErr = computeWindow_ (eWindow);
47
48 m_bIsInitialized = true;
49
50 return rErr;
51}

91Error_t CFft::compFft( complex_t *pfSpectrum, const float *pfInput )
92{
93 if (!m_bIsInitialized)
94 return Error_t::kNotInitializedError;
95 if (!pfInput || !pfSpectrum)
96 return Error_t::kFunctionInvalidArgsError;
97
98 // copy data to internal buffer
99 CVector::copy(m_pfProcessBuff, pfInput, m_iDataLength);
100 CVector::setZero(&m_pfProcessBuff[m_iDataLength], static_cast<long long>(m_iFftLength)-m_iDataLength);
101
102 // apply window function
103 if (m_ePrePostWindowOpt & kPreWindow)
104 CVector::mul_I(m_pfProcessBuff, m_pfWindowBuff, m_iDataLength);
105
106 // compute fft
107 LaszloFft::realfft_split(m_pfProcessBuff, m_iFftLength);
108
109 // copy data to output buffer
110 CVector::copy(pfSpectrum, m_pfProcessBuff, m_iFftLength);
111
112 return Error_t::kNoError;

136Error_t CFft::getMagnitude( float *pfMag, const complex_t *pfSpectrum ) const
137{
138 if (!m_bIsInitialized)
139 return Error_t::kNotInitializedError;
140
141 // re(0),re(1),re(2),...,re(size/2),im(size/2-1),...,im(1)
142 int iNyq = m_iFftLength>>1;
143
144 // no imaginary part at these bins
145 pfMag[0] = std::abs(pfSpectrum[0]);
146 pfMag[iNyq] = std::abs(pfSpectrum[iNyq]);
147
148 for (int i = 1; i < iNyq; i++)
149 {
150 int iImagIdx = m_iFftLength - i;
151 pfMag[i] = sqrtf(pfSpectrum[i]*pfSpectrum[i] + pfSpectrum[iImagIdx]*pfSpectrum[iImagIdx]);
152 }
153 return Error_t::kNoError;
154}

156Error_t CFft::getPhase( float *pfPhase, const complex_t *pfSpectrum ) const
157{
158 if (!m_bIsInitialized)
159 return Error_t::kNotInitializedError;
160
161 // re(0),re(1),re(2),...,re(size/2),im(size/2-1),...,im(1)
162 int iNyq = m_iFftLength>>1;
163
164 pfPhase[0] = m_Pi;
165 pfPhase[iNyq] = m_Pi;
166
167 for (int i = 1; i < iNyq; i++)
168 {
169 int iImagIdx = m_iFftLength - i;
170 if (pfSpectrum[i] == .0F && pfSpectrum[iImagIdx] != .0F)
171 pfPhase[i] = m_Pi2;
172 else
173 pfPhase[i] = atan2f (pfSpectrum[iImagIdx], pfSpectrum[i]);
174 }
175 return Error_t::kNoError;
176}

232void CFft::multiply_I(complex_t* pfFftSrc1Dest, const complex_t* pfFftSrc2) const
233{
234 int i = 0,
235 j = 0;
236
237 // re(0),re(1),re(2),...,re(size/2),im(size/2-1),...,im(1)
238 pfFftSrc1Dest[0] *= pfFftSrc2[0];
239 pfFftSrc1Dest[m_iFftLength >> 1] *= pfFftSrc2[m_iFftLength >> 1];
240
241 for (i = 1, j = m_iFftLength-1; i < m_iFftLength>>1; i++, j--)
242 {
243 float fTemp = pfFftSrc1Dest[i];
244
245 // real part of multiplication
246 pfFftSrc1Dest[i] *= pfFftSrc2[i];
247 pfFftSrc1Dest[i] -= pfFftSrc1Dest[j] * pfFftSrc2[j];
248
249 // imaginary part of multiplication
250 pfFftSrc1Dest[j] *= pfFftSrc2[i];
251 pfFftSrc1Dest[j] += fTemp * pfFftSrc2[j];
252 }
253
254 return;
255}

281Error_t CFft::computeWindow_( WindowFunction_t eWindow )
282{
283 int i;
284
285 // note that these windows are periodic, not symmetric
286 switch (eWindow)
287 {
288 case kWindowSine:
289 {
290 for (i = 0; i < m_iDataLength; i++)
291 {
292 m_pfWindowBuff[i] = sinf((i * m_Pi) / (m_iDataLength + 1));
293 }
294 break;
295 }
296 case kWindowHann:
297 {
298 for (i = 0; i < m_iDataLength; i++)
299 {
300 m_pfWindowBuff[i] = .5F * (1.F - cosf((i * 2.F * m_Pi) / (m_iDataLength + 1)));
301 }
302 break;
303 }
304 case kWindowHamming:
305 {
306 for (i = 0; i < m_iDataLength; i++)
307 {
308 m_pfWindowBuff[i] = .54F - .46F * cosf((i * 2.F * m_Pi) / (m_iDataLength + 1));
309 }
310 break;
311 }
312 default:
313 case kNumWindows:
314 {
315 return Error_t::kFunctionInvalidArgsError;
316 }
317 }
318
319 return Error_t::kNoError;
320}
