113int CKnn::classify(const float *pfQuery)
114{
115 if (!pfQuery || !m_bIsInitialized)
116 return kIllegalClassLabel;
117
118 // normalize
119 CVector::copy(m_pfQuery, pfQuery, m_iNumFeatures);
120 normalizeVector(m_pfQuery, m_iNumFeatures);
121
122 // compute distance to all training observations
123 for (auto n = 0; n < m_iNumObs; n++)
124 m_pfSortDist[n] = CVector::distEuclidean(pfQuery, m_ppfTrain[n], m_iNumFeatures);
125
126 // sort distances
127 CVector::sort_I(m_pfSortDist, m_piSortIdx, m_iNumObs);
128
129 // take care of weird scenarios
130 if (m_iK > m_iNumObs)
131 m_iK = m_iNumObs;
132
133 // create histogram
134 buildHistogram_(false);
135
136 // handle multiple maxima (first, use distances, then, reduce K)
137 int iK = m_iK; //!< remember m_iK
138 while (countMaxima_() > 1)
139 {
140 buildHistogram_(true);
141 m_iK--;
142 }
143 m_iK = iK;
144
145 return m_piHistLabel[0];
146}

148void CKnn::buildHistogram_(bool bUseDistance)
149{
150 int iNumLabelsInK = 0;
151
152 // init
153 CVector::setZero(m_pfHist, m_iK);
154 CVector::setZero(m_piHistCount, m_iK);
155 CVector::setValue(m_piHistLabel, kIllegalClassLabel, m_iK);
156
157 // set first candidate (lowest distance)
158 m_piHistLabel[0] = m_piClassLabels[m_piSortIdx[0]];
159 m_pfHist[0] += bUseDistance ? m_pfSortDist[0] : 1.F;
160 m_piHistCount[0]++;
161 iNumLabelsInK++;
162 for (auto k = 1; k < m_iK; k++)
163 {
164 bool bContinue = false;
165 for (auto l = 0; l < iNumLabelsInK; l++)
166 {
167 // check if current label k belongs to existing class
168 if (m_piHistLabel[l] == m_piClassLabels[m_piSortIdx[k]])
169 {
170 m_pfHist[l] += bUseDistance ? m_pfSortDist[k] : 1.F;
171 m_piHistCount[l]++;
172 bContinue = true;
173 break;
174 }
175 }
176 if (bContinue)
177 continue;
178
179 // create new hist class
180 m_piHistLabel[iNumLabelsInK] = m_piClassLabels[m_piSortIdx[k]];
181 m_pfHist[iNumLabelsInK] += bUseDistance ? m_pfSortDist[k] : 1.F;
182 m_piHistCount[iNumLabelsInK]++;
183 iNumLabelsInK++;
184 }
185
186 // distance-based metrics
187 if (bUseDistance)
188 {
189 // scale by number of entries in histogram
190 for (auto k = 0; k < iNumLabelsInK; k++)
191 if (m_piHistCount[k] > 0) m_pfHist[k] /= m_piHistCount[k];
192
193 // invert so that distance becomes similarity
194 float fMax = CVector::getMax(m_pfHist, iNumLabelsInK);
195 if (fMax > 0)
196 CVector::mulC_I(m_pfHist, -1.F / fMax, iNumLabelsInK);
197 CVector::addC_I(m_pfHist, 1.F, iNumLabelsInK);
198 }
199}

201int CKnn::countMaxima_()
202{
203 int iNumMax = 0;
204 float fMax = 0;
205 long long iMax = 0;
206
207 // find first max
208 CVector::findMax(m_pfHist, fMax, iMax, m_iK);
209 CUtil::swap(m_pfHist[iNumMax], m_pfHist[iMax]);
210 CUtil::swap(m_piHistLabel[iNumMax], m_piHistLabel[iMax]);
211 iNumMax++;
212 for (auto k = 1; k < m_iK; k++)
213 {
214 // check for others
215 if (m_pfHist[k] >= fMax)
216 {
217 CUtil::swap(m_pfHist[iNumMax], m_pfHist[k]);
218 CUtil::swap(m_piHistLabel[iNumMax], m_piHistLabel[k]);
219
220 iNumMax++;
221 }
222 }
223 return iNumMax;
