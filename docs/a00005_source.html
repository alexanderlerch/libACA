218{

350Error_t CSpectrogramIf::compMelSpectrogram(float** ppfMelSpectrogram, const MelSpectrogramConfig_t* pMelSpecConfig)
351{
352 if (!m_bIsInitialized)
353 return Error_t::kFunctionIllegalCallError;
354 if (!pMelSpecConfig)
355 return Error_t::kFunctionInvalidArgsError;
356 if (!ppfMelSpectrogram)
357 return Error_t::kFunctionInvalidArgsError;
358 if (!ppfMelSpectrogram[0])
359 return Error_t::kFunctionInvalidArgsError;
360
361 assert(m_pfProcessBuff);
362 assert(m_pfSpectrum);
363 assert(m_pCFft);
364
365 // initialize mel filterbank
366 generateMelFb_(pMelSpecConfig);
367 assert(m_ppfHMel);
368
369 long long iNumBlocks = m_pCBlockAudio->getNumBlocks();
370
371 for (auto n = 0; n < iNumBlocks; n++)
372 {
373 auto iLength = m_pCFft->getLength(CFft::kLengthMagnitude);
374
375 computeMagSpectrum_(iLength);
376
377 // copy to output buffer
378 for (auto k = 0; k < pMelSpecConfig->iNumMelBins; k++)
379 {
380 assert(m_ppfHMel[k]);
381 ppfMelSpectrogram[k][n] = CVectorFloat::mulScalar(m_ppfHMel[k], m_pfSpectrum, iLength);
382 }
383
384 if (pMelSpecConfig->bIsLogarithmic)
385 {
386 // convert amplitude to level(dB)
387 for (auto k = 0; k < pMelSpecConfig->iNumMelBins; k++)
388 ppfMelSpectrogram[k][n] = 20.F * std::log10(ppfMelSpectrogram[k][n] + 1e-12F);
389 }
390 }
391
392 destroyMelFb_(pMelSpecConfig);
393
394 return Error_t::kNoError;
395}

397Error_t CSpectrogramIf::generateMelFb_(const MelSpectrogramConfig_t* pMelSpecConfig)
398{
399 assert(pMelSpecConfig);
400
401 int iMagLength = m_pCFft->getLength(CFft::kLengthMagnitude);
402
403 // configuration check
404 if (pMelSpecConfig->fMinFreqInHz < 0)
405 return Error_t::kFunctionInvalidArgsError;
406 if (pMelSpecConfig->fMaxFreqInHz > m_fSampleRate / 2)
407 return Error_t::kFunctionInvalidArgsError;
408 if (pMelSpecConfig->iNumMelBins <= 0 || pMelSpecConfig->iNumMelBins > iMagLength)
409 return Error_t::kFunctionInvalidArgsError;
410
411 // allocate filter matrix and frequency matrix
412 m_pffcMel = new float[pMelSpecConfig->iNumMelBins + 2]; // +2 for lower and upper bound
413 m_ppfHMel = new float* [pMelSpecConfig->iNumMelBins];
414 for (auto k = 0; k < pMelSpecConfig->iNumMelBins; k++)
415 {
416 m_ppfHMel[k] = new float[iMagLength];
417 CVectorFloat::setZero(m_ppfHMel[k], iMagLength);
418 }
419
420 // compute center band frequencies
421 m_pffcMel[0] = CConversion::convertFreq2Mel(pMelSpecConfig->fMinFreqInHz);
422 m_pffcMel[pMelSpecConfig->iNumMelBins + 1] = CConversion::convertFreq2Mel(pMelSpecConfig->fMaxFreqInHz);
423 float fMelInc = (m_pffcMel[pMelSpecConfig->iNumMelBins + 1] - m_pffcMel[0]) / (pMelSpecConfig->iNumMelBins+1);
424 for (auto k = 1; k < pMelSpecConfig->iNumMelBins + 1; k++)
425 m_pffcMel[k] = m_pffcMel[k - 1] + fMelInc;
426 for (auto k = 0; k < pMelSpecConfig->iNumMelBins + 2; k++)
427 m_pffcMel[k] = CConversion::convertMel2Freq(m_pffcMel[k]);
428
429 float* pf_l = &m_pffcMel[0],
430 * pf_c = &m_pffcMel[1],
431 * pf_u = &m_pffcMel[2];
432
433 for (auto m = 0; m < pMelSpecConfig->iNumMelBins; m++)
434 {
435 float fFilterMax = 2.F / (pf_u[m] - pf_l[m]); //!< normalization
436
437 int iLowBin = 1 + static_cast<int>(m_pCFft->freq2bin(pf_l[m], m_fSampleRate));
438 int iCenterBin = 1 + static_cast<int>(m_pCFft->freq2bin(pf_c[m], m_fSampleRate));
439 int iUpBin = 1 + static_cast<int>(m_pCFft->freq2bin(pf_u[m], m_fSampleRate));
440
441 for (auto k = iLowBin; k < iCenterBin; k++)
442 m_ppfHMel[m][k] = fFilterMax * (CConversion::convertBin2Freq(static_cast<float>(k), (iMagLength - 1) * 2, m_fSampleRate) - pf_l[m]) / (pf_c[m] - pf_l[m]);
443 for (auto k = iCenterBin; k < iUpBin; k++)
444 m_ppfHMel[m][k] = fFilterMax * (pf_u[m] - CConversion::convertBin2Freq(static_cast<float>(k), (iMagLength - 1) * 2, m_fSampleRate)) / (pf_u[m] - pf_c[m]);
445
446 }
447
448 return Error_t::kNoError;
