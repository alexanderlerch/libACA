173Error_t CFeatureIf::compFeature1Dim(float* pfFeature)
174{
175 if (!m_bIsInitialized)
176 return Error_t::kFunctionIllegalCallError;
177 if (!pfFeature)
178 return Error_t::kFunctionInvalidArgsError;
179
180 assert(m_pfProcessBuff1);
181 assert(m_pfProcessBuff2);
182 assert(m_pCFft);
183 assert(m_pCBlockAudio);
184 assert(m_pCNormalize);
185 assert(m_pCFeature);
186
187 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
188
189 for (auto n = 0; n < iNumBlocks; n++)
190 {
191 // retrieve the next audio block
192 m_pCBlockAudio->getNextBlock(m_pfProcessBuff1);
193
194 // normalize if specified
195 if (m_pCNormalize)
196 m_pCNormalize->normalizeBlock(m_pfProcessBuff1, m_iBlockLength);
197
198 if (isFeatureSpectral_(m_pCFeature->getFeatureIdx()))
199 computeMagSpectrum_();
200
201 m_pCFeature->compFeature(&pfFeature[n], m_pfProcessBuff1);
202 }
203
204 return Error_t::kNoError;
205}

207Error_t CFeatureIf::compFeatureNDim(float** ppfFeature)
208{
209 if (!m_bIsInitialized)
210 return Error_t::kFunctionIllegalCallError;
211 if (!ppfFeature)
212 return Error_t::kFunctionInvalidArgsError;
213 if (!ppfFeature[0])
214 return Error_t::kFunctionInvalidArgsError;
215
216 assert(m_pfProcessBuff1);
217 assert(m_pfProcessBuff2);
218 assert(m_pCBlockAudio);
219 assert(m_pCNormalize);
220 assert(m_pCFeature);
221
222 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
223 auto iNumFeatures = m_pCFeature->getFeatureDimensions();
224
225 for (auto n = 0; n < iNumBlocks; n++)
226 {
227 // retrieve the next audio block
228 m_pCBlockAudio->getNextBlock(m_pfProcessBuff1);
229
230 // normalize if specified
231 if (m_pCNormalize)
232 m_pCNormalize->normalizeBlock(m_pfProcessBuff1, m_iBlockLength);
233
234 if (isFeatureSpectral_(m_pCFeature->getFeatureIdx()))
235 computeMagSpectrum_();
236
237 m_pCFeature->compFeature(m_pfProcessBuff2, m_pfProcessBuff1);
238
239 // copy to output buffer
240 for (auto k = 0; k < iNumFeatures; k++)
241 ppfFeature[k][n] = m_pfProcessBuff2[k];
242 }
243
244 return Error_t::kNoError;
245}
