66Error_t CCcf::compCcf(const float* pfInput1, const float* pfInput2, bool bNormalize)
67{
68 if (!pfInput1 || !pfInput2)
69 return Error_t::kFunctionInvalidArgsError;
70
71 if (!m_bIsInitialized)
72 return Error_t::kFunctionIllegalCallError;
73
74 // extract standard deviation for normalization
75 float afStd[2] = { 1.F, 1.F };
76 if (bNormalize)
77 {
78 float fPower = std::sqrt(CVectorFloat::mulScalar(pfInput1, pfInput1, m_iBlockLength));
79 afStd[0] = fPower > 0 ? fPower : 1.F;
80 fPower = std::sqrt(CVectorFloat::mulScalar(pfInput2, pfInput2, m_iBlockLength));
81 afStd[1] = fPower > 0 ? fPower : 1.F;
82 }
83
84 // compute the FFTs
85 m_pCFft->compFft(m_apfData[0], pfInput1);
86 m_pCFft->compFft(m_apfData[1], pfInput2);
87
88 // conjugate complex multiply
89 CVectorFloat::mulC_I(m_apfData[1], static_cast<float>(m_iFftLength), m_iFftLength);
90 m_pCFft->conjugate_I(m_apfData[1]);
91 m_pCFft->multiply_I(m_apfData[0], m_apfData[1]);
92
93 // normalization
94 CVectorFloat::mulC_I(m_apfData[0], 1.F / (afStd[0] * afStd[1]), m_iFftLength);
95
96 // inverse Fft
97 m_pCFft->compInvFft(m_apfData[0], m_apfData[0]);
98
99 // copy results
100 CVectorFloat::copy(m_apfData[1], &m_apfData[0][m_iFftLength - m_iBlockLength + 1], m_iBlockLength - 1);
101 CVectorFloat::copy(&m_apfData[1][m_iBlockLength - 1], m_apfData[0], m_iBlockLength);
102
103 m_bWasProcessed = true;
104 return Error_t::kNoError;
105}
