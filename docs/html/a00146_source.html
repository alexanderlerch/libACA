105 \param fA4InHz tuning (reference) frequency in Hz

108 static float convertFreq2Midi(float fInHz, float fA4InHz = 440.F)
109 {
110 assert(fInHz > 0);
111 assert(fA4InHz > 0);
112 return 69.F + 12.F * std::log2(fInHz / fA4InHz);
113 };
