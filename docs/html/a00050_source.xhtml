59Error_t CNmf::compNmf(CNmfResult* pCNmfResult, float** ppfInput)
60{
61
62 if (!pCNmfResult || !ppfInput)
63 return Error_t::kFunctionInvalidArgsError;
64
65 if (!m_bIsInitialized || !pCNmfResult->isInitialized())
66 return Error_t::kFunctionIllegalCallError;
67
68 enum Cost_t
69 {
70 kStartCost,
71 kPrevCost,
72 kCurrCost,
73
74 kNumCosts
75 };
76 float afCost[kNumCosts] = { 0 };
77
78 for (int k = 0; k < m_iMaxIter; k++)
79 {
80 afCost[kCurrCost] = runNmfIter(pCNmfResult, ppfInput);
81
82 assert(!std::isnan(afCost[kCurrCost]));
83
84 if (k > 0)
85 {
86 if (std::abs((afCost[kCurrCost] - afCost[kPrevCost]) / (afCost[kStartCost] - afCost[kCurrCost] + m_kMinOffset)) < .001F)
87 break;
88 }
89 else
90 afCost[kStartCost] = afCost[kCurrCost];
91
92 afCost[kPrevCost] = afCost[kCurrCost];
93 }
94 return Error_t::kNoError;

97float CNmf::runNmfIter(CNmfResult* pCNmfResult, float** ppfInput)
98{
99 float** ppfW = pCNmfResult->getMatPointer(CNmfResult::kW);
100 float** ppfH = pCNmfResult->getMatPointer(CNmfResult::kH);
101 float** ppfXHat = pCNmfResult->getMatPointer(CNmfResult::kXHat);
102
103 int iRank = pCNmfResult->getMatCols(CNmfResult::kW);
104 int iNumObs = pCNmfResult->getMatCols(CNmfResult::kH);
105 int iNumBins = pCNmfResult->getMatRows(CNmfResult::kW);
106
107 const bool bUpdateH = true,
108 bUpdateW = true;
109
110 // compute current estimate
111 CMatrix::mulMatMat(ppfXHat, ppfW, ppfH, iNumBins, iRank, iRank, iNumObs);
112 CMatrix::addC_I(ppfXHat, m_kMinOffset, iNumBins, iNumObs);
113
114 //m_ppfX
115 CMatrix::copy(m_ppfX, ppfInput, iNumBins, iNumObs);
116 CMatrix::div_I(m_ppfX, ppfXHat, iNumBins, iNumObs);
117
118 // update H
119 if (bUpdateH)
120 {
121 CMatrix::transpose(m_ppfTransp, ppfW, iNumBins, iRank);
122
123 // denominator
124 CMatrix::mulMatMat(m_ppfDenom, m_ppfTransp, m_ppfOnes, iRank, iNumBins, iNumBins, iNumObs);
125 CMatrix::addC_I(m_ppfDenom, m_fSparsity + m_kMinOffset, iRank, iNumObs);
126
127 // numerator
128 CMatrix::mulMatMat(m_ppfNum, m_ppfTransp, m_ppfX, iRank, iNumBins, iNumBins, iNumObs);
129 CMatrix::div_I(m_ppfNum, m_ppfDenom, iRank, iNumObs);
130 CMatrix::mul_I(ppfH, m_ppfNum, iRank, iNumObs);
131
132 // clean small values
133 CMatrix::setZeroBelowThresh(ppfH, iRank, iNumObs, m_kMinOffset);
134 }
135
136 // update W
137 if (bUpdateW)
138 {
139 CMatrix::transpose(m_ppfTransp, ppfH, iRank, iNumObs);
140
141 // denominator
142 CMatrix::mulMatMat(m_ppfDenom, m_ppfOnes, m_ppfTransp, iNumBins, iNumObs, iNumObs, iRank);
143 CMatrix::addC_I(m_ppfDenom, m_kMinOffset, iNumBins, iRank);
144
145 // numerator
146 CMatrix::mulMatMat(m_ppfNum, m_ppfX, m_ppfTransp, iNumBins, iNumObs, iNumObs, iRank);
147 CMatrix::div_I(m_ppfNum, m_ppfDenom, iNumBins, iRank);
148 CMatrix::mul_I(ppfW, m_ppfNum, iNumBins, iRank);
149
150 // normalize
151 CMatrix::vecnorm_I(ppfW, iNumBins, iRank);
152
153 // clean small values
154 CMatrix::setZeroBelowThresh(ppfW, iNumBins, iRank, m_kMinOffset);
155 }
156
157 // compute error
158 CMatrix::mulMatMat(ppfXHat, ppfW, ppfH, iNumBins, iRank, iRank, iNumObs);
159 CMatrix::addC_I(ppfXHat, m_kMinOffset, iNumBins, iNumObs);
160
161 // deal with sparsity
162 CMatrix::addC_I(ppfXHat, m_fSparsity * CMatrix::getNorm(ppfXHat, iNumBins, iNumObs), iNumBins, iNumObs);
163
164 return CMatrix::calcKlDivergence(ppfInput, ppfXHat, iNumBins, iNumObs);
165}
