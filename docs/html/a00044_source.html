54Error_t CKmeans::compKmeans(int* piResult, float** ppfFeatures)
55{
56 if (!m_bIsInitialized)
57 return Error_t::kFunctionIllegalCallError;
58 if (!piResult || !ppfFeatures)
59 return Error_t::kFunctionInvalidArgsError;
60 if (!ppfFeatures[0])
61 return Error_t::kFunctionInvalidArgsError;
62
63 // init: randomly selected data points as cluster means
64 initClusterMeans_(ppfFeatures);
65
66 // assign observations to clusters
67 assignClusterLabels_(piResult, ppfFeatures);
68
69 // iterate
70 for (auto i = 0; i < m_iMaxIter; i++)
71 {
72 // copy state to prev state
73 CMatrix::copy(m_appfClusterMeans[kPrev], m_appfClusterMeans[kCurr], m_iK, m_iNumFeatures);
74
75 // update cluster means
76 computeClusterMeans_(ppfFeatures, piResult);
77
78 // re-init empty clusters
79 reinitClusterMeans_(ppfFeatures);
80
81 // assign observations to clusters
82 assignClusterLabels_(piResult, ppfFeatures);
83
84 // check for change
85 if (checkConverged_())
86 break;
87 }
88
89 return Error_t::kNoError;

126void CKmeans::computeClusterMeans_(float** ppfFeatures, const int* piResult)
127{
128 CMatrix::setZero(m_appfClusterMeans[kCurr], m_iK, m_iNumFeatures);
129 //CVector::setZero(m_piClusterSize, m_iK);
130 for (auto n = 0; n < m_iNumObs; n++)
131 {
132 for (auto v = 0; v < m_iNumFeatures; v++)
133 m_appfClusterMeans[kCurr][piResult[n]][v] += ppfFeatures[v][n];
134 }
135 for (auto k = 0; k < m_iK; k++)

137 if (!m_piClusterSize[k])

139 CVector::mulC_I(m_appfClusterMeans[kCurr][k], 1.F / m_piClusterSize[k], m_iNumFeatures);

143bool CKmeans::checkConverged_()
144{
145 float fSum = 0;
146 int k = 0;
147 CVector::copy(m_pfProc, m_appfClusterMeans[kCurr][k], m_iNumFeatures);
148 CVector::sub_I(m_pfProc, m_appfClusterMeans[kPrev][k], m_iNumFeatures);
149 while ((fSum += CVector::getSum(m_pfProc, m_iNumFeatures, true)) <= 0 && k < m_iK - 1)
150 {
151 k++;
152 CVector::copy(m_pfProc, m_appfClusterMeans[kCurr][k], m_iNumFeatures);
153 CVector::sub_I(m_pfProc, m_appfClusterMeans[kPrev][k], m_iNumFeatures);
154 }
155 if (fSum <= 0)
156 return true;
157
158 return false;

161void CKmeans::assignClusterLabels_(int* piResult, float** ppfFeatures)
162{
163 CVector::setZero(m_piClusterSize, m_iK);
164 for (auto n = 0; n < m_iNumObs; n++)

166 float fMin = 0;

170 // compute distance to all training observations
171 for (auto k = 0; k < m_iK; k++)
