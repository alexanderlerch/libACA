179Error_t CNoveltyIf::compNovelty(float* pfNovelty, bool* pbIsOnset)
180{
181 if (!m_bIsInitialized)
182 return Error_t::kFunctionIllegalCallError;
183 if (!pfNovelty)
184 return Error_t::kFunctionInvalidArgsError;
185
186 assert(m_pfProcessBuff1);
187 assert(m_pfProcessBuff2);
188 assert(m_pCFft);
189 assert(m_pCBlockAudio);
190 assert(m_pCNormalize);
191 assert(m_pCNovelty);
192
193 const float fSmoothLpLenInS = 0.07F;
194 const float fThreshLpLenInS = 0.14F;
195 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
196
197 float *pfThreshold = 0; //!< memory allocation is ok since we compDtw the whole signal at once
198 CVector::alloc(pfThreshold, iNumBlocks);
199
200 for (auto n = 0; n < iNumBlocks; n++)
201 {
202 // retrieve the next audio block
203 m_pCBlockAudio->getNextBlock(m_pfProcessBuff1);
204
205 // normalize if specified
206 if (m_pCNormalize)
207 m_pCNormalize->normalizeBlock(m_pfProcessBuff1, m_iBlockLength);
208
209 computeMagSpectrum_();
210
211 m_pCNovelty->compNovelty(&pfNovelty[n], m_pfProcessBuff1);
212 }
213
214 // normalize
215 float fMax = CVectorFloat::getMax(pfNovelty, iNumBlocks, true);
216 if (fMax > 0)
217 CVectorFloat::mulC_I(pfNovelty, 1 / fMax, iNumBlocks);
218
219 // smoothing with moving average
220 m_pCLpFilter->reset();
221 m_pCLpFilter->setFilterParam(m_pCLpFilter->calcFilterLength(fSmoothLpLenInS, m_fSampleRate / m_iHopLength));
222 m_pCLpFilter->filtfilt(pfNovelty, pfNovelty, iNumBlocks);
223
224 // HWR
225 CVectorFloat::setZeroBelowThresh(pfNovelty, iNumBlocks);
226
227 // threshold computation
228 m_pCLpFilter->reset();
229 m_pCLpFilter->setFilterParam(m_pCLpFilter->calcFilterLength(fThreshLpLenInS, m_fSampleRate / m_iHopLength));
230 m_pCLpFilter->filtfilt(pfThreshold, pfNovelty, iNumBlocks);
231 CVectorFloat::addC_I(pfThreshold, .4F * CVectorFloat::getMean(&pfNovelty[1], iNumBlocks - 1), iNumBlocks);
232 CVectorFloat::sub_I(pfThreshold, pfNovelty, iNumBlocks);
233 CVectorFloat::mulC_I(pfThreshold, -1.F, iNumBlocks);
234 CVectorFloat::setZeroBelowThresh(pfThreshold, iNumBlocks);
235
236 // peak picking
237 if (pbIsOnset)
238 {
239 CVector::setValue(pbIsOnset, false, iNumBlocks);
240 CVectorFloat::findPeaks(pbIsOnset, pfThreshold, iNumBlocks, 0.F);
241 }
242
243 // clean up memory
244 CVector::free(pfThreshold);
245
246 return Error_t::kNoError;
247}
