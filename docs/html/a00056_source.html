98Error_t CPitchIf::create(CPitchIf*& pCInstance, PitchExtractors_t ePitchIdx, const std::string& strAudioFilePath, int iBlockLength, int iHopLength)
99{
100 if (strAudioFilePath.empty())
101 return Error_t::kFunctionInvalidArgsError;
102 if (iBlockLength <= 2 || !CUtil::isPowOf2(iBlockLength))
103 return Error_t::kFunctionInvalidArgsError;
104 if (iHopLength <= 0 || iHopLength > iBlockLength)
105 return Error_t::kFunctionInvalidArgsError;
106
107 pCInstance = new CPitchFromFile(ePitchIdx, strAudioFilePath, iBlockLength, iHopLength);
108
109
110 return Error_t::kNoError;

186Error_t CPitchIf::compF0(float* pfPitch)
187{
188 if (!m_bIsInitialized)
189 return Error_t::kFunctionIllegalCallError;
190 if (!pfPitch)
191 return Error_t::kFunctionInvalidArgsError;
192
193 assert(m_pfProcessBuff1);
194 assert(m_pCBlockAudio);
195 assert(m_pCNormalize);
196 assert(m_pCPitch);
197
198 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
199
200 for (auto n = 0; n < iNumBlocks; n++)
201 {
202 // retrieve the next audio block
203 m_pCBlockAudio->getNextBlock(m_pfProcessBuff1);
204
205 // normalize if specified
206 if (m_pCNormalize)
207 m_pCNormalize->normalizeBlock(m_pfProcessBuff1, m_iBlockLength);
208
209 if (isPitchExtractorSpectral_(m_pCPitch->getPitchExtractorIdx()))
210 {
211 assert(m_pfProcessBuff2);
212 assert(m_pCFft);
213 computeMagSpectrum_();
214 }
215
216 pfPitch[n] = m_pCPitch->compF0(m_pfProcessBuff1);
217 }
218
219 return Error_t::kNoError;
220}

293Error_t CPitchIf::init_(PitchExtractors_t ePitchIdx)
294{
295 if (isPitchExtractorSpectral_(ePitchIdx))
296 {
297 // initialize FFT and fft buffer
298 m_pCFft = new CFft();
299 m_pCFft->init(m_iBlockLength);
300 // allocate processing memory
301 CVector::alloc(m_pfProcessBuff1, m_pCFft->getLength(CFft::kLengthFft));
302 CVector::alloc(m_pfProcessBuff2, m_pCFft->getLength(CFft::kLengthFft));
303 CPitchFromBlockIf::create(m_pCPitch, ePitchIdx, m_pCFft->getLength(CFft::kLengthMagnitude), m_fSampleRate);
304 }
305 else
306 {
307 // allocate processing memory
308 CVector::alloc(m_pfProcessBuff1, m_iBlockLength);
309 CPitchFromBlockIf::create(m_pCPitch, ePitchIdx, m_iBlockLength, m_fSampleRate);
310 }
311 m_bIsInitialized = true;
312
313 return Error_t::kNoError;
