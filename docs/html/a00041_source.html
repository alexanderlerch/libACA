30Error_t CFft::init( int iBlockLength, int iZeroPadFactor, WindowFunction_t eWindow /*= kWindowHann*/, Windowing_t eWindowing /*= kPreWindow*/ )
31{
32 Error_t rErr = Error_t::kNoError;
33
34 // sanity check
35 if (iZeroPadFactor <= 0)
36 return Error_t::kFunctionInvalidArgsError;
37
38 // clean up
39 reset();
40
41 // make sure the fft length is a power of two
42 int iInternalBlockLength = CUtil::isPowOf2(iBlockLength) ? iBlockLength : CUtil::nextPowOf2(iBlockLength);
43 m_iDataLength = iBlockLength;
44 m_iFftLength = iInternalBlockLength * iZeroPadFactor;
45
46 m_ePrePostWindowOpt = eWindowing;
47
48 rErr = allocMemory_ ();
49 if (rErr != Error_t::kNoError)
50 return rErr;
51
52 rErr = computeWindow_ (eWindow);
53
54 m_bIsInitialized = true;
55
56 return rErr;
57}

142Error_t CFft::getMagnitude( float *pfMag, const complex_t *pfSpectrum ) const
143{
144 if (!m_bIsInitialized)
145 return Error_t::kNotInitializedError;
146
147 // re(0),re(1),re(2),...,re(size/2),im(size/2-1),...,im(1)
148 int iNyq = m_iFftLength>>1;
149
150 // no imaginary part at these bins
151 pfMag[0] = std::abs(pfSpectrum[0]);
152 pfMag[iNyq] = std::abs(pfSpectrum[iNyq]);
153
154 for (int i = 1; i < iNyq; i++)
155 {
156 int iImagIdx = m_iFftLength - i;
157 pfMag[i] = sqrtf(pfSpectrum[i]*pfSpectrum[i] + pfSpectrum[iImagIdx]*pfSpectrum[iImagIdx]);
158 }
159 return Error_t::kNoError;
160}

162Error_t CFft::getPhase( float *pfPhase, const complex_t *pfSpectrum ) const
163{

166
167 // re(0),re(1),re(2),...,re(size/2),im(size/2-1),...,im(1)
168 int iNyq = m_iFftLength>>1;
169
170 pfPhase[0] = m_Pi;
171 pfPhase[iNyq] = m_Pi;
172
173 for (int i = 1; i < iNyq; i++)
174 {
175 int iImagIdx = m_iFftLength - i;
176 if (pfSpectrum[i] == .0F && pfSpectrum[iImagIdx] != .0F)
177 pfPhase[i] = m_Pi2;
178 else
179 pfPhase[i] = atan2f (pfSpectrum[iImagIdx], pfSpectrum[i]);
180 }
181 return Error_t::kNoError;

238void CFft::multiply_I(complex_t* pfFftSrc1Dest, const complex_t* pfFftSrc2) const
239{
240 int i = 0,
241 j = 0;
242
243 // re(0),re(1),re(2),...,re(size/2),im(size/2-1),...,im(1)
244 pfFftSrc1Dest[0] *= pfFftSrc2[0];
245 pfFftSrc1Dest[m_iFftLength >> 1] *= pfFftSrc2[m_iFftLength >> 1];
246
247 for (i = 1, j = m_iFftLength-1; i < m_iFftLength>>1; i++, j--)
248 {
249 float fTemp = pfFftSrc1Dest[i];
250
251 // real part of multiplication
252 pfFftSrc1Dest[i] *= pfFftSrc2[i];
253 pfFftSrc1Dest[i] -= pfFftSrc1Dest[j] * pfFftSrc2[j];
254
255 // imaginary part of multiplication
256 pfFftSrc1Dest[j] *= pfFftSrc2[i];
257 pfFftSrc1Dest[j] += fTemp * pfFftSrc2[j];
258 }
259
260 return;
261}

288Error_t CFft::computeWindow_( WindowFunction_t eWindow )
289{
290 int i;
291
292 // note that these windows are periodic, not symmetric
293 switch (eWindow)
294 {
295 case kWindowSine:
296 {
297 for (i = 0; i < m_iDataLength; i++)
298 {
299 m_pfWindowBuff[i] = sinf((i * m_Pi) / (m_iDataLength + 1));
300 }
301 break;
302 }
303 case kWindowHann:
304 {
305 for (i = 0; i < m_iDataLength; i++)
306 {
307 m_pfWindowBuff[i] = .5F * (1.F - cosf((i * 2.F * m_Pi) / (m_iDataLength + 1)));
308 }
309 break;
310 }
311 case kWindowHamming:
312 {
313 for (i = 0; i < m_iDataLength; i++)
314 {
315 m_pfWindowBuff[i] = .54F - .46F * cosf((i * 2.F * m_Pi) / (m_iDataLength + 1));
316 }
317 break;
318 }
319 default:
320 case kNumWindows:
321 {
322 return Error_t::kFunctionInvalidArgsError;
323 }
324 }
325
326 return Error_t::kNoError;
327}
