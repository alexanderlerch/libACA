242 CVector::alloc(pfTmpBuff, iLengthOfBuffer + 2 * static_cast<long long>(iFilterLength));
243
244 this->process(&pfTmpBuff[iFilterLength], pfInput, iLengthOfBuffer);
245
246 // tail
247 for (auto i = 0; i < iFilterLength; i++)
248 {
249 float fZero = 0.F;
250 this->process(&pfTmpBuff[iLengthOfBuffer + iFilterLength + i], &fZero, 1);
251 }
252
253 // reverse tail

259
260 for (auto i = iLengthOfBuffer + iFilterLength - 1; i >= iFilterLength; i--)
261 this->process(&pfOutput[i-iFilterLength], &pfTmpBuff[i], 1);
262
263 this->reset();
264 this->setFilterParam(iFilterLength);

267 }
