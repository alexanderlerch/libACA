641 {

643 assert(iNumRows == iNumCols);
644 assert(ppfSrcDest);
645 assert(ppfSrcDest[0]);

649 float** ppfEye = 0;

651 double dDet = 1;
652
653 alloc(ppfTmp, iNumRows, iNumCols);
654 copy(ppfTmp, ppfSrcDest, iNumRows, iNumCols);
655
656 alloc(ppfEye, iNumRows, iNumCols);
657 setEye(ppfEye, iNumRows, iNumCols);

659 if (ppfSrcDest[0][0] == 0)
660 {
661 i = 1;
662 while (i < iNumRows)
663 {
664 if (ppfSrcDest[i][0] != 0)
665 {
666 swapRowCol(ppfSrcDest, 0, i, iNumCols);
667 swapRowCol(ppfEye, 0, i, iNumCols);
668 dDet *= -1;
669 break;
670 }
671 i++;
672 }
673 }

675 dDet *= ppfSrcDest[0][0];
676
677 CVectorFloat::mulC_I(ppfEye[0], 1.F / ppfSrcDest[0][0], iNumCols);

679

685 dDet = 0;
686

688 {

690 CVectorFloat::subW_I(ppfSrcDest[i], ppfSrcDest[j], ppfSrcDest[i][j], iNumCols);
691 j++;
692 }
693
694 if (ppfSrcDest[i][i] != 0)
695 {
696 dDet *= ppfSrcDest[i][i];
697 CVectorFloat::mulC_I(ppfEye[i], 1.F / ppfSrcDest[i][i], iNumCols);
698 CVectorFloat::mulC_I(ppfSrcDest[i], 1.F / ppfSrcDest[i][i], iNumCols);
699 }
700
701 if (ppfSrcDest[i][i] == 0)
702 {
