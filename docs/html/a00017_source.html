172Error_t CFeatureIf::compFeature1Dim(float* pfFeature)
173{
174 if (!m_bIsInitialized)
175 return Error_t::kFunctionIllegalCallError;
176 if (!pfFeature)
177 return Error_t::kFunctionInvalidArgsError;
178
179 assert(m_pfProcessBuff1);
180 assert(m_pfProcessBuff2);
181 assert(m_pCFft);
182 assert(m_pCBlockAudio);
183 assert(m_pCNormalize);
184 assert(m_pCFeature);
185
186 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
187
188 for (auto n = 0; n < iNumBlocks; n++)
189 {
190 // retrieve the next audio block
191 m_pCBlockAudio->getNextBlock(m_pfProcessBuff1);
192
193 // normalize if specified
194 if (m_pCNormalize)
195 m_pCNormalize->normalizeBlock(m_pfProcessBuff1, m_iBlockLength);
196
197 if (isFeatureSpectral_(m_pCFeature->getFeatureIdx()))
198 computeMagSpectrum_();
199
200 m_pCFeature->compFeature(&pfFeature[n], m_pfProcessBuff1);
201 }
202
203 return Error_t::kNoError;
204}

206Error_t CFeatureIf::compFeatureNDim(float** ppfFeature)
207{
208 if (!m_bIsInitialized)
209 return Error_t::kFunctionIllegalCallError;
210 if (!ppfFeature)
211 return Error_t::kFunctionInvalidArgsError;
212 if (!ppfFeature[0])
213 return Error_t::kFunctionInvalidArgsError;
214
215 assert(m_pfProcessBuff1);
216 assert(m_pfProcessBuff2);
217 assert(m_pCBlockAudio);
218 assert(m_pCNormalize);
219 assert(m_pCFeature);
220
221 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
222 auto iNumFeatures = m_pCFeature->getFeatureDimensions();
223
224 for (auto n = 0; n < iNumBlocks; n++)
225 {
226 // retrieve the next audio block
227 m_pCBlockAudio->getNextBlock(m_pfProcessBuff1);
228
229 // normalize if specified
230 if (m_pCNormalize)
231 m_pCNormalize->normalizeBlock(m_pfProcessBuff1, m_iBlockLength);
232
233 if (isFeatureSpectral_(m_pCFeature->getFeatureIdx()))
234 computeMagSpectrum_();
235
236 m_pCFeature->compFeature(m_pfProcessBuff2, m_pfProcessBuff1);
237
238 // copy to output buffer
239 for (auto k = 0; k < iNumFeatures; k++)
240 ppfFeature[k][n] = m_pfProcessBuff2[k];
241 }
242
243 return Error_t::kNoError;
