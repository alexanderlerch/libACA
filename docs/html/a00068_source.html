103void CViterbi::compProbability_(float** ppfPEmission)
104{
105 // initialize
106 for (auto m = 0; m < m_iNumStates; m++)
107 {
108 CVectorFloat::setValue(m_ppfProb[m], 1, m_iNumObs);
109 m_ppfProb[m][0] = ppfPEmission[m][0] * m_pfStart[m];
110 }
111
112
113 // compute probability matrix and store backtracking path
114 for (int n = 1; n < m_iNumObs; n++)
115 {
116 for (int m = 0; m < m_iNumStates; m++)
117 {
118 // find max of preceding times trans prob
119 float fMaxProb = 0;
120 for (auto s = 0; s < m_iNumStates; s++)
121 {
122 float fProbsm = m_ppfProb[s][n - 1] * m_ppfTransProb[s][m];
123 if (fProbsm > fMaxProb)
124 {
125 fMaxProb = fProbsm;
126 m_ppfProb[m][n] = fMaxProb;
127 m_ppiPathIdx[m][n] = s;
128 }
129 }
130 m_ppfProb[m][n] *= ppfPEmission[m][n];
131 }
132 }

135void CViterbi::compLogLikelihood_(float** ppfPEmission)
136{
137 // convert trans prob to log
138 for (auto m = 0; m < m_iNumStates; m++)
139 for (auto s = 0; s < m_iNumStates; s++)
140 m_ppfTransProb[m][s] = std::log(m_ppfTransProb[m][s] + m_kLogMin);

144 m_ppfProb[m][0] = std::log(ppfPEmission[m][0]+ m_kLogMin) + std::log(m_pfStart[m]+ m_kLogMin);
145
146
147 // compute probability matrix and store backtracking path
148 for (int n = 1; n < m_iNumObs; n++)
149 {
150 for (int m = 0; m < m_iNumStates; m++)

152 // find max of preceding times trans prob

154 for (auto s = 0; s < m_iNumStates; s++)
