51 /*! performs the InstFreq computation on subsequent spectra
52 \param pfInstFreq output vector containing the estimated frequencies (to be written), length iFftLength/2+1
53 \param pfSpectrum current (complex) spectrum

56 Error_t process(float *pfInstFreq, const CFft::complex_t* pfSpectrum)
57 {
58 if (!pfInstFreq || !pfSpectrum)
59 return Error_t::kFunctionInvalidArgsError;
60
61 // get current phase
62 m_pCFft->getPhase(m_apfPhase[kPhaseCurr], pfSpectrum);
63
64 // calc instantaneous frequency
65 CVector::sub_I(m_apfPhase[kPhasePrev], m_apfPhase[kPhaseCurr], m_iPhaseLength);
66 CVector::add_I(m_apfPhase[kPhasePrev], m_pfOmega, m_iPhaseLength);
67 CVector::mulC_I(m_apfPhase[kPhasePrev], -1.0F, m_iPhaseLength);
68
69 princArg_(m_apfPhase[kPhasePrev]);
70
71 CVector::add_I(m_apfPhase[kPhasePrev], m_pfOmega, m_iPhaseLength);
72 CVector::copy(pfInstFreq, m_apfPhase[kPhasePrev], m_iPhaseLength);
73 CVector::mulC_I(pfInstFreq, static_cast<float>(m_fSampleRate / (m_iHopLength * 2. * M_PI)), m_iPhaseLength);
74
75 // remember phase for next call but avoid copying
76 CUtil::swap(m_apfPhase[kPhasePrev], m_apfPhase[kPhaseCurr]);
