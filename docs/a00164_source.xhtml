70 /*! generates a sawtooth

72 \param fFreqInHz frequency of sinusoidal

74 \param iLength number of frames to be generated

80 {
81 if (!pfOutBuf)
82 return Error_t::kFunctionInvalidArgsError;
83
84 float fIncr = 2*fAmplitude / fSampleRateInHz * fFreqInHz;
85 pfOutBuf[0] = 0;
86 for (int i = 1; i < iLength; i++)

88 pfOutBuf[i] = fmodf(pfOutBuf[i-1] + fIncr + fAmplitude, 2*fAmplitude) - fAmplitude;
