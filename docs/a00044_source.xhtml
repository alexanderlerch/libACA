107Error_t CGammatone::process(float *pfOut, const float *pfIn, long long iNumSamples)
108{
109 if (!pfOut || !pfIn || iNumSamples <= 0)
110 return Error_t::kFunctionInvalidArgsError;
111
112 if (!m_bIsInitialized)
113 return Error_t::kFunctionIllegalCallError;
114
115#ifdef ACA_USE_DOUBLE
116 for (auto i = 0; i < iNumSamples; i++)
117 {
118 double fTmp = pfIn[i];
119 for (auto c = 0; c < kNumFilters; c++)
120 m_apCFilter[c]->process(&fTmp, &fTmp, 1);
121 pfOut[i] = static_cast<float>(fTmp);
122 }
123#else
124 CVector::copy(pfOut, pfIn, iNumSamples);
125 for (auto c = 0; c < kNumFilters; c++)
126 m_apCFilter[c]->process(pfOut, pfOut, iNumSamples);
127#endif // ACA_USE_DOUBLE
128
129 return Error_t::kNoError;

147void CGammatone::calcFilterCoeffs_(int iOrder)
148{
149 assert(m_fSampleRate > 0);
150 assert(m_fFreqCenter >= 0);
151
152 const double f2Pi = 2 * M_PI;
153 const double fEarQ = 9.26449;
154 const double fBandwidth = 24.7;
155 const double B = 1.019 * f2Pi * std::pow(std::pow(m_fFreqCenter / fEarQ, iOrder) + std::pow(fBandwidth, iOrder), 1. / iOrder);
156
157 double fArg = f2Pi * m_fFreqCenter / m_fSampleRate;
158 double fCos = std::cos(fArg);
159 double fSin = std::sin(fArg);
160 double fExp = std::exp(B / m_fSampleRate);
161 double fSqrtA = 2 * std::sqrt(3. + std::pow(2., 1.5F));
162 double fSqrtS = 2 * std::sqrt(3. - std::pow(2., 1.5F));
163 double fGain = 0;
164
165 for (auto c = 0; c < kNumFilters; c++)
166 {
167 m_aafCoeffB[c][0] = 1. / m_fSampleRate;
168 m_aafCoeffB[c][2] = 0.;
169
170 m_aafCoeffA[c][0] = 1.;
171 m_aafCoeffA[c][1] = -2. * fCos / fExp;
172 m_aafCoeffA[c][2] = std::exp(-2. * B / m_fSampleRate);
173 }
174
175 m_aafCoeffB[0][1] = -(2. / m_fSampleRate * fCos / fExp + fSqrtA / m_fSampleRate * fSin / fExp) / 2.;
176 m_aafCoeffB[1][1] = -(2. / m_fSampleRate * fCos / fExp - fSqrtA / m_fSampleRate * fSin / fExp) / 2.;
177 m_aafCoeffB[2][1] = -(2. / m_fSampleRate * fCos / fExp + fSqrtS / m_fSampleRate * fSin / fExp) / 2.;
178 m_aafCoeffB[3][1] = -(2. / m_fSampleRate * fCos / fExp - fSqrtS / m_fSampleRate * fSin / fExp) / 2.;
179
180 // apply gain to first cascade
181 {
182 fSqrtA *= .5;
183 fSqrtS *= .5;
184
185 double fScale = 2 * 2 * 2 * 2 / (m_fSampleRate * m_fSampleRate * m_fSampleRate * m_fSampleRate);
186 std::complex<double> num1(-std::cos(2. * fArg), -std::sin(2 * fArg));
187 std::complex<double> num2(std::cos(fArg) * std::exp(-B / m_fSampleRate), std::sin(fArg) * std::exp(-B / m_fSampleRate));
188 std::complex<double> denom(-2. / std::exp(2. * B / m_fSampleRate) + 2. / fExp + std::cos(2 * fArg) * (2. / fExp - 2.), std::sin(2 * fArg) * (2. / fExp - 2.));
189
190 auto numerator = fScale *
191 (num1 + num2 * (fCos - fSqrtS * fSin)) *
192 (num1 + num2 * (fCos + fSqrtS * fSin)) *
193 (num1 + num2 * (fCos - fSqrtA * fSin)) *
194 (num1 + num2 * (fCos + fSqrtA * fSin));
195 auto denominator = denom * denom * denom * denom;
196 fGain = std::abs(numerator / denominator);
197 }
198 for (auto k = 0; k < kNumCoeffs; k++)
199 m_aafCoeffB[0][k] /= fGain;
200
201 for (auto c = 0; c < kNumFilters; c++)
202 m_apCFilter[c]->init(m_aafCoeffB[c], m_aafCoeffA[c], 3);
203}

214
215 float fArg = f2Pi * m_fFreqCenter / m_fSampleRate;
216 float fCos = std::cos(fArg);
217 float fSin = std::sin(fArg);
218 float fExp = std::exp(B / m_fSampleRate);
219 float fSqrtA = 2 * std::sqrt(3.F + std::pow(2.F, 1.5F));
220 float fSqrtS = 2 * std::sqrt(3.F - std::pow(2.F, 1.5F));
221 float fGain = 0;
222
223 for (auto c = 0; c < kNumFilters; c++)
224 {
225 m_aafCoeffB[c][0] = 1.F / m_fSampleRate;
226 m_aafCoeffB[c][2] = 0.F;
227
228 m_aafCoeffA[c][0] = 1.F;
229 m_aafCoeffA[c][1] = -2.F * fCos / fExp;
230 m_aafCoeffA[c][2] = std::exp(-2.F * B / m_fSampleRate);
231 }
232
233 m_aafCoeffB[0][1] = -(2.F / m_fSampleRate * fCos / fExp + fSqrtA / m_fSampleRate * fSin / fExp) / 2.F;
234 m_aafCoeffB[1][1] = -(2.F / m_fSampleRate * fCos / fExp - fSqrtA / m_fSampleRate * fSin / fExp) / 2.F;
235 m_aafCoeffB[2][1] = -(2.F / m_fSampleRate * fCos / fExp + fSqrtS / m_fSampleRate * fSin / fExp) / 2.F;
236 m_aafCoeffB[3][1] = -(2.F / m_fSampleRate * fCos / fExp - fSqrtS / m_fSampleRate * fSin / fExp) / 2.F;
237
238 // apply gain to first cascade
239 {
240 fSqrtA *= .5F;
241 fSqrtS *= .5F;
242
243 float fScale = 2 * 2 * 2 * 2 / (m_fSampleRate * m_fSampleRate * m_fSampleRate * m_fSampleRate);
244 std::complex<float> num1(-std::cos(2.F * fArg), -std::sin(2 * fArg));
245 std::complex<float> num2(std::cos(fArg) * std::exp(-B / m_fSampleRate), std::sin(fArg) * std::exp(-B / m_fSampleRate));
246 std::complex<float> denom(-2.F / std::exp(2.F * B / m_fSampleRate) + 2.F / fExp + std::cos(2 * fArg) * (2.F / fExp - 2.F), std::sin(2 * fArg) * (2.F / fExp - 2.F));
247
248 auto numerator = fScale *
249 (num1 + num2 * (fCos - fSqrtS * fSin)) *
250 (num1 + num2 * (fCos + fSqrtS * fSin)) *
251 (num1 + num2 * (fCos - fSqrtA * fSin)) *
252 (num1 + num2 * (fCos + fSqrtA * fSin));
253 auto denominator = denom * denom * denom * denom;
254 fGain = std::abs(numerator / denominator);
255 }
256 CVector::mulC_I(m_aafCoeffB[0], 1.F / fGain, kNumCoeffs);
257
258 for (auto c = 0; c < kNumFilters; c++)
259 m_apCFilter[c]->init(m_aafCoeffB[c], m_aafCoeffA[c], 3);

435Error_t CGammaToneFbIf::process(float **ppfOut, const float *pfIn /*= 0*/, long long iNumSamples /*= 0*/)
436{
437 if (!m_bIsInitialized)
438 return Error_t::kFunctionIllegalCallError;
439 if (!ppfOut)
440 return Error_t::kFunctionInvalidArgsError;
441 if (!ppfOut[0])
442 return Error_t::kFunctionInvalidArgsError;
443
444 // special case of block by block processing - no normalization, no blocking
445 if (pfIn)
446 {
447 assert(iNumSamples > 0);
448 for (auto i = 0; i < m_iNumBands; i++)
449 {
450 assert(ppfOut[i]);
451 m_ppCGammatone[i]->process(ppfOut[i], pfIn, iNumSamples);
452 }
453 return Error_t::kNoError;
454 }
455
456 assert(m_pCBlockAudio);
457
458 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
459
460 for (auto n = 0; n < iNumBlocks; n++)
461 {
462 // retrieve the next audio block
463 int iNumFrames = m_pCBlockAudio->getNextBlock(m_pfProcBuff);
464
465 // normalize if specified
466 if (m_pCNormalize)
467 m_pCNormalize->normalizeBlock(m_pfProcBuff, m_iBlockLength);
468
469 for (auto i = 0; i < m_iNumBands; i++)
470 {
471 assert(ppfOut[i]);
472 m_ppCGammatone[i]->process(&ppfOut[i][n * m_iBlockLength], m_pfProcBuff, iNumFrames);
473 }
474 }
475
476 return Error_t::kNoError;
477}

514inline float CGammaToneFbIf::compMidFreqs_(float fFreqLow, float fFreqHigh, int k) const
515{
516 const float fEarQ = 9.26449F;
517 const float fBandwidth = 24.7F;
518
519 return -(fEarQ * fBandwidth) + std::exp((m_iNumBands - k) * (-std::log(fFreqHigh + fEarQ * fBandwidth) + std::log(fFreqLow + fEarQ * fBandwidth)) / m_iNumBands) * (fFreqHigh + fEarQ * fBandwidth);
520}
