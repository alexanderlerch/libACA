41 // note we store this transposed - that's easier for the distance computation

57 for (auto f = 0; f < m_iNumFeatures; f++)
58 {
59 m_pfNormSub[f] = CVector::getMin(ppfTrainFeatures[f], m_iNumObs);
60 m_pfNormScale[f] = CVector::getMax(ppfTrainFeatures[f], m_iNumObs) - m_pfNormSub[f];
61 if (m_pfNormScale[f] != 0)
62 m_pfNormScale[f] = 1.F / m_pfNormScale[f];
63 }
64 }
65 else
66 {

68 CVector::setZero(m_pfNormSub, m_iNumFeatures);
69 }
70
71 // normalize features
72 for (auto n = 0; n < m_iNumObs; n++)
73 {
74 CVector::sub_I(m_ppfTrain[n], m_pfNormSub, m_iNumFeatures);
75 CVector::mul_I(m_ppfTrain[n], m_pfNormScale, m_iNumFeatures);
76 }
77
78 m_bIsInitialized = true;

126int CKnn::classify(const float* pfQuery)
127{
128 if (!pfQuery || !m_bIsInitialized)
129 return kIllegalClassLabel;
130
131 // normalize
132 CVector::copy(m_pfQuery, pfQuery, m_iNumFeatures);
133 CVector::sub_I(m_pfQuery, m_pfNormSub, m_iNumFeatures);
134 CVector::mul_I(m_pfQuery, m_pfNormScale, m_iNumFeatures);
135
136 // compute distance to all training observations
137 for (auto n = 0; n < m_iNumObs; n++)
138 m_pfDist[n] = CVector::distEuclidean(pfQuery, m_ppfTrain[n], m_iNumFeatures);
139
140 // sort distances
141 CVector::sort_I(m_pfDist, m_piSortIdx, m_iNumObs);
142
143 // take care of weird scenarios
144 if (m_iK > m_iNumObs)
145 m_iK = m_iNumObs;
146
147 // create histogram
148 buildHistogram_(false);
149
150 // handle multiple maxima (first, use distances, then, reduce K)
151 int iK = m_iK; //!< remember m_iK
152 while (countMaxima_() > 1)
153 {
154 buildHistogram_(true);
155 m_iK--;
156 }
157 m_iK = iK;
158
159 return m_piHistLabel[0];
160}

162void CKnn::buildHistogram_(bool bUseDistance)
163{
164 int iNumLabelsInK = 0;
165 CVector::setValue(m_piHistLabel, kIllegalClassLabel, m_iK);
166 m_piHistLabel[0] = m_piClassLabels[m_piSortIdx[0]];
167 m_pfHist[0] += 1;
168 iNumLabelsInK++;
169 for (auto k = 1; k < m_iK; k++)
170 {
171 bool bContinue = false;
172 for (auto l = 0; l < iNumLabelsInK; l++)
173 {
174 if (m_piHistLabel[l] == m_piClassLabels[m_piSortIdx[k]])
175 {
176 m_pfHist[l] += bUseDistance ? m_pfDist[m_piSortIdx[k]] : 1.F;
177 bContinue = true;
178 break;
179 }
180 }
181 if (bContinue)
182 continue;
183 m_piHistLabel[iNumLabelsInK] = m_piClassLabels[m_piSortIdx[k]];
184 m_pfHist[iNumLabelsInK] += 1;
185 iNumLabelsInK++;
186 }
187}

189int CKnn::countMaxima_()
190{

192 float fMax = 0;
193 long long iMax = 0;
194 CVector::findMax(m_pfHist, fMax, iMax, m_iK);
195 CUtil::swap(m_pfHist[iNumMax], m_pfHist[iMax]);
196 CUtil::swap(m_piHistLabel[iNumMax], m_piHistLabel[iMax]);
197 iNumMax++;
198 for (auto k = 1; k < m_iK; k++)
199 {
200 if (m_pfHist[k] >= fMax)
201 {
202 CUtil::swap(m_pfHist[iNumMax], m_pfHist[k]);
203 CUtil::swap(m_piHistLabel[iNumMax], m_piHistLabel[k]);
204
205 iNumMax++;
206 }
207 }
208 return iNumMax;
