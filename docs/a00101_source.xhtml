62Error_t CGmm::compGmm(CGmmResult* pCResult, float** ppfFeatures)
63{
64 if (!pCResult || !ppfFeatures)
65 return Error_t::kFunctionInvalidArgsError;
66 if (!ppfFeatures[0])
67 return Error_t::kFunctionInvalidArgsError;
68 if (!m_bisInitialized)

93void CGmm::initState_(float** ppfFeatures, CGmmResult* pCCurrState)
94{
95 // generate some noise
96 CSynthesis::genNoise(m_apfProc[0], m_iK);
97 CVector::addC_I(m_apfProc[0], 1.F, m_iK);
98 CVector::mulC_I(m_apfProc[0], (m_iNumObs - 1) / 2.F, m_iK);
99
100 CPca::compCov(m_appfSigma[0], ppfFeatures, m_iNumFeatures, m_iNumObs);
101 for (auto k = 0; k < m_iK; k++)
102 {
103 int iIdx = CUtil::float2int<int>(m_apfProc[0][k]);
104 assert(iIdx > 0 && iIdx < m_iNumObs);
105
106 // set sigma
107 pCCurrState->setSigma(k, m_appfSigma[0]);
108
109 // set mean
110 for (auto v = 0; v < m_iNumFeatures; v++)
111 pCCurrState->setMu(k, v, ppfFeatures[v][iIdx]);
112
113 // set prior
114 pCCurrState->setPrior(k, 1.F / m_iK);
115 }

118void CGmm::compProbabilities_(float** ppfFeatures, CGmmResult* pCCurrState)
119{
120 // compute gaussian per cluster per observation
121 for (auto k = 0; k < m_iK; k++)
122 {
123 pCCurrState->getSigma(m_appfSigma[0], k);
124 float fDet = CMatrix::det(m_appfSigma[0], m_iNumFeatures, m_iNumFeatures);
125 if (fDet < 1e-30F)
126 {
127 CVector::setZero(m_ppfProb[k], m_iNumObs);
128 continue;
129 }
130
131 float fNorm = static_cast<float>(1. / std::sqrt(std::pow(2. * M_PI, m_iNumFeatures) * fDet));
132 CMatrix::inv_I(m_appfSigma[0], m_iNumFeatures, m_iNumFeatures);
133
134 for (auto n = 0; n < m_iNumObs; n++)
135 {
136 for (auto v = 0; v < m_iNumFeatures; v++)
137 m_apfProc[0][v] = ppfFeatures[v][n] - pCCurrState->getMu(k, v);
138
139 CMatrix::mulMatColVec(m_apfProc[1], m_appfSigma[0], m_apfProc[0], m_iNumFeatures, m_iNumFeatures);
140
141 m_ppfProb[k][n] = fNorm * std::exp(-.5F * CVector::mulScalar(m_apfProc[0], m_apfProc[1], m_iNumFeatures));
142 }
143 CVector::mulC_I(m_ppfProb[k], pCCurrState->getPrior(k), m_iNumObs);
144 }
145
146 // norm over clusters
147 for (auto n = 0; n < m_iNumObs; n++)
148 {
149 float fSum = CMatrix::getSumCol(m_ppfProb, n, m_iK);
150
151 if (fSum > 0)
152 CMatrix::mulColC_I(m_ppfProb, 1.F / fSum, n, m_iK);
153 }
154}

156void CGmm::updateState_(float** ppfFeatures, CGmmResult* pCCurrState)
157{
158 for (auto k = 0; k < m_iK; k++)
159 {
160 // update priors
161 pCCurrState->setPrior(k, CVector::getMean(m_ppfProb[k], m_iNumObs));
162
163 // update means
164 float fNorm = CVector::getSum(m_ppfProb[k], m_iNumObs);
165 if (fNorm < 1e-30F)
166 fNorm = 1.F;
167 for (auto v = 0; v < m_iNumFeatures; v++)
168 pCCurrState->setMu(k, v, CVector::mulScalar(ppfFeatures[v], m_ppfProb[k], m_iNumObs) / fNorm);
169
170 // update sigma
171 CMatrix::setZero(m_appfSigma[0], m_iNumFeatures, m_iNumFeatures);
172 for (auto n = 0; n < m_iNumObs; n++)
173 {
174 for (auto v = 0; v < m_iNumFeatures; v++)
175 m_apfProc[0][v] = ppfFeatures[v][n] - pCCurrState->getMu(k, v);
176
177 CMatrix::mulColVecRowVec(m_appfSigma[1], m_apfProc[0], m_apfProc[0], m_iNumFeatures, m_iNumFeatures);
178 CMatrix::mulC_I(m_appfSigma[1], m_ppfProb[k][n], m_iNumFeatures, m_iNumFeatures);
179 CMatrix::add_I(m_appfSigma[0], m_appfSigma[1], m_iNumFeatures, m_iNumFeatures);
180 }
181 fNorm = CVector::getSum(m_ppfProb[k], m_iNumObs);
182 if (fNorm < 1e-30F)
183 fNorm = 1.F;
184 CMatrix::mulC_I(m_appfSigma[0], 1.F / fNorm, m_iNumFeatures, m_iNumFeatures);
185 pCCurrState->setSigma(k, m_appfSigma[0]);
186 }

189bool CGmm::checkConverged_(CGmmResult* pCCurrState)
190{
191 float fSum = 0;
192 for (auto k = 0; k < m_iK; k++)
193 {
194 for (auto v = 0; v < m_iNumFeatures; v++)
195 fSum += std::abs(pCCurrState->getMu(k, v) - PrevState.getMu(k, v));
196 }
197
198 if (fSum/m_iK <= 1e-20F)
199 return true;
200
201 return false;

223float CGmmResult::getProb(const float* pfQuery)
224{
225 float fProb = 0;
226 for (auto k = 0; k < m_iK; k++)
227 {
228 float fDet = CMatrix::det(m_apppfSigma[kNormal][k], m_iNumFeatures, m_iNumFeatures);
229 if (fDet < 1e-30F)
230 continue;
231 float fNorm = static_cast<float>(1. / std::sqrt(std::pow(2. * M_PI, m_iNumFeatures) * fDet));
232
233 CVector::copy(m_apfProc[0], pfQuery, m_iNumFeatures);

235
236 CMatrix::mulMatColVec(m_apfProc[1], m_apppfSigma[kInv][k], m_apfProc[0], m_iNumFeatures, m_iNumFeatures);

238 fProb += this->getPrior(k) * fNorm * std::exp(-.5F * CVector::mulScalar(m_apfProc[0], m_apfProc[1], m_iNumFeatures));
