102void CViterbi::compProbability_(const float *const *const ppfPEmission)
103{
104 // initialize
105 for (auto m = 0; m < m_iNumStates; m++)
106 {
107 CVector::setValue(m_ppfProb[m], 1.F, m_iNumObs);
108 m_ppfProb[m][0] = ppfPEmission[m][0] * m_pfStart[m];
109 }
110
111 // compute probability matrix and store backtracking path
112 for (int n = 1; n < m_iNumObs; n++)
113 {
114 for (int m = 0; m < m_iNumStates; m++)
115 {
116 // find max of preceding prob times trans prob
117 float fMaxProb = 0;
118 for (auto s = 0; s < m_iNumStates; s++)
119 {
120 float fProbsm = m_ppfProb[s][n - 1] * m_ppfTransProb[s][m];
121 if (fProbsm > fMaxProb)
122 {
123 fMaxProb = fProbsm;
124 m_ppfProb[m][n] = fMaxProb;
125 m_ppiPathIdx[m][n] = s;
126 }
127 }
128 m_ppfProb[m][n] *= ppfPEmission[m][n];
129 }
130 }

133void CViterbi::compLogLikelihood_(const float *const *const ppfPEmission)
134{
135 // convert trans prob to log
136 for (auto m = 0; m < m_iNumStates; m++)
137 for (auto s = 0; s < m_iNumStates; s++)
138 m_ppfTransProb[m][s] = std::log(m_ppfTransProb[m][s] + m_kLogMin);
139
140 // initialize
141 for (auto m = 0; m < m_iNumStates; m++)
142 m_ppfProb[m][0] = std::log(ppfPEmission[m][0] + m_kLogMin) + std::log(m_pfStart[m] + m_kLogMin);
143
144 // compute probability matrix and store backtracking path
145 for (int n = 1; n < m_iNumObs; n++)
146 {
147 for (int m = 0; m < m_iNumStates; m++)
148 {
149 // find max of preceding prob times trans prob
150 float fMaxProb = -std::numeric_limits<float>::max();
151 for (auto s = 0; s < m_iNumStates; s++)
152 {
153 float fProbsm = m_ppfProb[s][n - 1] + m_ppfTransProb[s][m];
154 if (fProbsm > fMaxProb)
155 {
156 fMaxProb = fProbsm;
157 m_ppfProb[m][n] = fMaxProb;
158 m_ppiPathIdx[m][n] = s;
159 }
160 }
161 m_ppfProb[m][n] += std::log(ppfPEmission[m][n] + m_kLogMin);
