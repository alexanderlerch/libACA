179Error_t CFeatureIf::compFeature1Dim(float* pfFeature)
180{
181 if (!m_bIsInitialized)
182 return Error_t::kFunctionIllegalCallError;
183 if (!pfFeature)
184 return Error_t::kFunctionInvalidArgsError;
185
186 assert(m_pfProcessBuff1);
187 assert(m_pfProcessBuff2);
188 assert(m_pCFft);
189 assert(m_pCBlockAudio);
190 assert(m_pCNormalize);
191 assert(m_pCFeature);
192
193 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
194
195 for (auto n = 0; n < iNumBlocks; n++)
196 {
197 // retrieve the next audio block
198 m_pCBlockAudio->getNextBlock(m_pfProcessBuff1);
199
200 // normalize if specified
201 if (m_pCNormalize)
202 m_pCNormalize->normalizeBlock(m_pfProcessBuff1, m_iBlockLength);
203
204 if (isFeatureSpectral_(m_pCFeature->getFeatureIdx()))
205 computeMagSpectrum_();
206
207 m_pCFeature->compFeature(&pfFeature[n], m_pfProcessBuff1);
208 }
209
210 return Error_t::kNoError;
211}

213Error_t CFeatureIf::compFeatureNDim(float** ppfFeature)
214{
215 if (!m_bIsInitialized)
216 return Error_t::kFunctionIllegalCallError;
217 if (!ppfFeature)
218 return Error_t::kFunctionInvalidArgsError;
219 if (!ppfFeature[0])
220 return Error_t::kFunctionInvalidArgsError;
221
222 assert(m_pfProcessBuff1);
223 assert(m_pfProcessBuff2);
224 assert(m_pCBlockAudio);
225 assert(m_pCNormalize);
226 assert(m_pCFeature);
227
228 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
229 auto iNumFeatures = m_pCFeature->getFeatureDimensions();
230
231 for (auto n = 0; n < iNumBlocks; n++)
232 {
233 // retrieve the next audio block
234 m_pCBlockAudio->getNextBlock(m_pfProcessBuff1);
235
236 // normalize if specified
237 if (m_pCNormalize)
238 m_pCNormalize->normalizeBlock(m_pfProcessBuff1, m_iBlockLength);
239
240 if (isFeatureSpectral_(m_pCFeature->getFeatureIdx()))
241 computeMagSpectrum_();
242
243 m_pCFeature->compFeature(m_pfProcessBuff2, m_pfProcessBuff1);
244
245 // copy to output buffer
246 for (auto k = 0; k < iNumFeatures; k++)
247 ppfFeature[k][n] = m_pfProcessBuff2[k];
248 }
249
250 return Error_t::kNoError;
