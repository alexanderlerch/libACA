30Error_t CChordFromBlockIf::compChordProb(float *pfChordProb, const float *pfIn)
31{
32 // compute pitch chroma
33 m_pCFeatureExtractor->compFeature(m_pfPitchChroma, pfIn);
34
35 // check if zero
36 if (CVector::getSum(m_pfPitchChroma, kNumPitchClasses) <= 1e-20F)
37 {
38 CVector::setZero(pfChordProb, CChordIf::kNumChords);
39 pfChordProb[CChordIf::kNoChord] = 1.F;
40 return Error_t::kNoError;
41 }
42
43 // get chord probs with template
44 CMatrix::mulMatColvec(pfChordProb, m_ppfTemplateMatrix, m_pfPitchChroma, CChordIf::kNumChords, kNumPitchClasses);
45
46 assert(CVector::getSum(pfChordProb, CChordIf::kNumChords) > 0);
47
48 // normalize to probability of 1
49 CVector::mulC_I(pfChordProb, 1.F / CVector::getSum(pfChordProb, CChordIf::kNumChords), CChordIf::kNumChords);
50
51 return Error_t::kNoError;
52}

72void CChordFromBlockIf::genTemplateMatrix_()
73{
74 assert(m_ppfTemplateMatrix);
75 assert(m_ppfTemplateMatrix[0]);
76
77 const int iNumChordPitches = 3;
78 const int aiMajorIndices[3] = { 0,4,7 };
79 const int aiMinorIndices[3] = { 0,3,7 };
80
81 // simple template with equally weighted chord pitches, zero otherwise
82 for (auto c = 0; c < kNumPitchClasses; c++)
83 {
84 for (auto p = 0; p < iNumChordPitches; p++)
85 {
86 m_ppfTemplateMatrix[c][(c + aiMajorIndices[p]) % 12] = 1.F / iNumChordPitches;
87 m_ppfTemplateMatrix[c + kNumPitchClasses][(c + aiMinorIndices[p]) % 12] = 1.F / iNumChordPitches;
88 }
89 }
90 CVector::addC_I(m_ppfTemplateMatrix[CChordIf::kNoChord], 1.F / kNumPitchClasses, kNumPitchClasses);
91}
