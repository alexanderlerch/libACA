62Error_t CGmm::compGmm(CGmmResult *pCResult, const float *const *const ppfFeatures)
63{
64 if (!pCResult || !ppfFeatures)
65 return Error_t::kFunctionInvalidArgsError;
66 if (!ppfFeatures[0])
67 return Error_t::kFunctionInvalidArgsError;
68 if (!m_bisInitialized)
69 return Error_t::kFunctionIllegalCallError;
70
71 // init: randomly selected data points as cluster means
72 initState_(ppfFeatures, pCResult);
73
74 // iterate
75 for (auto i = 0; i < m_iMaxIter; i++)
76 {
77 // copy state to prev state
78 PrevState = *pCResult;
79
80 // update cluster means
81 compProbabilities_(ppfFeatures, pCResult);
82
83 // assign observations to clusters
84 updateState_(ppfFeatures, pCResult);
85
86 // check for change
87 if (checkConverged_(pCResult))
88 break;
89 }
90 return Error_t::kNoError;
91}

93void CGmm::initState_(const float *const *const ppfFeatures, CGmmResult *pCCurrState)
94{
95 // generate some noise
96 CSynthesis::genNoise(m_apfProc[0], m_iK);
97 CVector::mulC_I(m_apfProc[0], m_iNumObs - 1.F, m_iK);
98
99 CPca::compCov(m_appfSigma[0], ppfFeatures, m_iNumFeatures, m_iNumObs);
100 for (auto k = 0; k < m_iK; k++)
101 {
102 int iIdx = CUtil::float2int<int>(m_apfProc[0][k]);
103 assert(iIdx >= 0 && iIdx < m_iNumObs);
104
105 // set sigma
106 pCCurrState->setSigma(k, m_appfSigma[0]);
107
108 // set mean
109 for (auto v = 0; v < m_iNumFeatures; v++)
110 pCCurrState->setMu(k, v, ppfFeatures[v][iIdx]);
111
112 // set prior
113 pCCurrState->setPrior(k, 1.F / m_iK);
114 }
115}

117void CGmm::compProbabilities_(const float *const *const ppfFeatures, CGmmResult *pCCurrState)
118{
119 // compute gaussian per cluster per observation
120 for (auto k = 0; k < m_iK; k++)
121 {
122 pCCurrState->getSigma(m_appfSigma[0], k);
123 float fDet = CMatrix::det(m_appfSigma[0], m_iNumFeatures, m_iNumFeatures);
124 if (fDet < 1e-30F)
125 {
126 CVector::setZero(m_ppfProb[k], m_iNumObs);
127 continue;
128 }
129
130 float fNorm = static_cast<float>(1. / std::sqrt(std::pow(2. * M_PI, m_iNumFeatures) * fDet));
131 CMatrix::inv_I(m_appfSigma[0], m_iNumFeatures, m_iNumFeatures);
132
133 for (auto n = 0; n < m_iNumObs; n++)
134 {
135 for (auto v = 0; v < m_iNumFeatures; v++)
136 m_apfProc[0][v] = ppfFeatures[v][n] - pCCurrState->getMu(k, v);
137
138 CMatrix::mulMatColvec(m_apfProc[1], m_appfSigma[0], m_apfProc[0], m_iNumFeatures, m_iNumFeatures);
139
140 m_ppfProb[k][n] = fNorm * std::exp(-.5F * CVector::mulScalar(m_apfProc[0], m_apfProc[1], m_iNumFeatures));
141 }
142 CVector::mulC_I(m_ppfProb[k], pCCurrState->getPrior(k), m_iNumObs);
143 }
144
145 // norm over clusters
146 for (auto n = 0; n < m_iNumObs; n++)
147 {
148 float fSum = CMatrix::getSumCol(m_ppfProb, n, m_iK);
149
150 if (fSum > 0)
151 CMatrix::mulColC_I(m_ppfProb, 1.F / fSum, n, m_iK);
152 }
153}

155void CGmm::updateState_(const float *const *const ppfFeatures, CGmmResult *pCCurrState)
156{
157 for (auto k = 0; k < m_iK; k++)
158 {
159 // update priors
160 pCCurrState->setPrior(k, CVector::getMean(m_ppfProb[k], m_iNumObs));
161
162 // update means
163 float fNorm = CVector::getSum(m_ppfProb[k], m_iNumObs);
164 if (fNorm < 1e-30F)
165 fNorm = 1.F;
166 for (auto v = 0; v < m_iNumFeatures; v++)
167 pCCurrState->setMu(k, v, CVector::mulScalar(ppfFeatures[v], m_ppfProb[k], m_iNumObs) / fNorm);
168
169 // update sigma
170 CMatrix::setZero(m_appfSigma[0], m_iNumFeatures, m_iNumFeatures);
171 for (auto n = 0; n < m_iNumObs; n++)
172 {
173 for (auto v = 0; v < m_iNumFeatures; v++)
174 m_apfProc[0][v] = ppfFeatures[v][n] - pCCurrState->getMu(k, v);
175
176 CMatrix::mulColvecRowvec(m_appfSigma[1], m_apfProc[0], m_apfProc[0], m_iNumFeatures, m_iNumFeatures);
177 CMatrix::mulC_I(m_appfSigma[1], m_ppfProb[k][n], m_iNumFeatures, m_iNumFeatures);
178 CMatrix::add_I(m_appfSigma[0], m_appfSigma[1], m_iNumFeatures, m_iNumFeatures);
179 }
180 fNorm = CVector::getSum(m_ppfProb[k], m_iNumObs);
181 if (fNorm < 1e-30F)
182 fNorm = 1.F;
183 CMatrix::mulC_I(m_appfSigma[0], 1.F / fNorm, m_iNumFeatures, m_iNumFeatures);
184 pCCurrState->setSigma(k, m_appfSigma[0]);
185 }

188bool CGmm::checkConverged_(CGmmResult *pCCurrState)
189{
190 float fSum = 0;
191 for (auto k = 0; k < m_iK; k++)
192 {
193 for (auto v = 0; v < m_iNumFeatures; v++)
194 fSum += std::abs(pCCurrState->getMu(k, v) - PrevState.getMu(k, v));
195 }
196
197 if (fSum / m_iK <= 1e-20F)
198 return true;
199
200 return false;
201}

222float CGmmResult::getProb(const float *pfQuery)
223{
224 float fProb = 0;
225 for (auto k = 0; k < m_iK; k++)
226 {
227 float fDet = CMatrix::det(m_apppfSigma[kNormal][k], m_iNumFeatures, m_iNumFeatures);
228 if (fDet < 1e-30F)
229 continue;
230 float fNorm = static_cast<float>(1. / std::sqrt(std::pow(2. * M_PI, m_iNumFeatures) * fDet));
231
232 CVector::copy(m_apfProc[0], pfQuery, m_iNumFeatures);
233 CVector::sub_I(m_apfProc[0], m_ppfMu[k], m_iNumFeatures);
234
235 CMatrix::mulMatColvec(m_apfProc[1], m_apppfSigma[kInv][k], m_apfProc[0], m_iNumFeatures, m_iNumFeatures);
236
237 fProb += this->getPrior(k) * fNorm * std::exp(-.5F * CVector::mulScalar(m_apfProc[0], m_apfProc[1], m_iNumFeatures));
238 }
239
240 return fProb;
241}
