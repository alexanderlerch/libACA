20 /* --------------------------------------------------------------------------------
21 * s7 itself is based on the types and functions in this file, so the first place to look for examples
22 * is s7.c. There are also a few variations on a REPL at the end of s7.html. s7test.scm
23 * is a regression test for s7 -- it still turns up a few problems. More tests are certainly welcome!
24 * Extended examples of s7 usage are:
25 *
26 * Snd: ftp://ccrma-ftp.stanford.edu/pub/Lisp/snd-14.tar.gz (a sound editor)
27 * which includes:
28 * libxm: libxm.tar.gz (X, Motif, Gtk, Cairo, and openGL bindings)
29 * sndlib: sndlib.tar.gz (sound file, audio port, and CLM bindings plus an optimizer (run))
30 *
31 * Common Music by Rick Taube: http://camil.music.uiuc.edu/Software/grace/downloads/cm3.tar.gz (composition)
32 * which can use sndlib -- see Snd's grfsnd.html or the cmdist archives for details
33 *
34 *
35 * s7 (Scheme) variables:
36 *
37 * *features* a list of symbols describing what is currently available (initially '(s7)).
38 * *vector-print-length* how many elements of a vector are printed (initially 8)
39 * *load-path* a list of directory names that "load" searches for Scheme input files (initially '())
40 * *load-hook* hook called before a file is loaded; takes a function of one arg, the name of the file.
41 * *error-hook* hook called upon error; takes a function of two args,
42 * the error type (a symbol), and the info about it (a list).
43 * *unbound-variable-hook* hook called when an unbound symbol is accessed.
44 * *#readers* #... readers
45 * *gc-stats* #t to turn on GC statistics
46 * __func__ equivalent to C's __func__. Data about the current function.
47 *
48 * s7 constants:
49 *
50 * most-positive-fixnum
51 * most-negative-fixnum integer limits (the names come from Common Lisp)
52 * pi 3.1415...
53 * *stdin*, *stdout*, *stderr* default IO ports
54 *
55 * s7 non-standard functions:
56 *
57 * provided? checks the *features* list for a symbol
58 * provide adds a symbol to the *features* list
59 * port-line-number current line during loading
60 * port-filename current file name during loading
61 * pair-line-number location of a piece of code read from a file
62 * gc calls the GC. If its argument is #f, the GC is turned off
63 * call-with-exit just like call/cc but without the ability to jump back into a context
64 * continuation? #t if its argument is a continuation (as opposed to an ordinary procedure)
65 * procedure-documentation doc string associated with a procedure
66 * procedure-arity a list describing the arglist of a function: '(required-args optional-args rest-arg)
67 * aritable? #t if a function can be called with a given number of arguments
68 * procedure-source returns the source (a list) of a procedure
69 * help tries to find a help string associated with its argument
70 * symbol-calls if profiling is enabled, returns the number of times its argument (a symbol) has been called
71 * macro? returns #t is its argument is a macro or a symbol whose value is a macro
72 * with-baffle limit scope of continuations
73 *
74 * and various others mentioned at the start of s7.c -- nearly every Scheme implementation includes
75 * stuff like logior, sinh, read-line, format, define*, etc. See also the start of s7.c for choices
76 * such as multiprecision arithmetic, multidimensional vectors, initial heap and stack size, etc.
77 *
78 * The functions map, for-each, length, reverse, copy, and fill! are generic.
79 *
80 * I think s7 has built-in support for srfi-0 (cond-expand), srfi-6 (basic string ports), srfi-8 (receive), srfi-17 (generalized-set!),
81 * srfi-28 (format, also nearly all of srfi-48), srfi-30 (block comments),
82 * srfi-88 (keywords), and srfi-89 (define*). It also supports the functionality of many others
83 * but under a slightly different syntax: srfi-69 (hash-tables), srfi-16 (define*), srfi-25 (multidimensional
84 * arrays). srfi-98 would be trivial to add, and exists in snd as getenv.
85 * The srfi-1 (lists) and srfi-60 (bitwise ops) reference implementations can be loaded as is.

209 /* any s7_pointer object held in C (as a local variable for example) needs to be
210 * protected from garbage collection if there is any chance the GC may run without
211 * an existing Scheme-level reference to that object. s7_gc_protect places the
212 * object in a vector that the GC always checks, returning the object's location
213 * in that table. s7_gc_unprotect and s7_gc_unprotect_at unprotect the object
214 * (remove it from the vector). s7_gc_unprotect_at uses the location passed
215 * to it, whereas s7_gc_unprotect scans the vector to find the object.
216 * s7_gc_protected_at returns the object at the given location.
217 *
218 * You can turn the GC on and off via s7_gc_on.
219 *
220 * There is a built-in lag between the creation of a new object and its first possible GC
221 * (the lag time is set indirectly by GC_TEMPS_SIZE in s7.c), so you don't need to worry about
222 * very short term temps such as the arguments to s7_cons in:
223 *
224 * s7_cons(s7, s7_make_real(s7, 3.14),
225 * s7_cons(s7, s7_make_integer(s7, 123),
226 * s7_nil(s7)));

616 /* s7_make_function creates a Scheme function object from the s7_function 'fnc'.
617 * Its name (for s7_describe_object) is 'name', it requires 'required_args' arguments,
618 * can accept 'optional_args' other arguments, and if 'rest_arg' is true, it accepts
619 * a "rest" argument (a list of all the trailing arguments). The function's documentation
620 * is 'doc'.
621 *
622 * s7_define_function is the same as s7_make_function, but it also adds 'name' (as a symbol) to the
623 * global (top-level) environment, with the function as its value. For example, the Scheme
624 * function 'car' is essentially:
625 *
626 * s7_pointer g_car(s7_scheme *sc, s7_pointer args)
627 * {return(s7_car(sc, s7_car(sc, args)));}
628 *
629 * then bound to the name "car":
630 *
631 * s7_define_function(sc, "car", g_car, 1, 0, false, "(car obj)");
632 * one required arg, no optional arg, no "rest" arg
633 *
634 * s7_is_function returns true if its argument is a function defined in this manner.
635 * s7_apply_function applies the function (the result of s7_make_function) to the arguments.
636 *
637 * s7_define_macro defines a Scheme macro; its arguments are not evaluated (unlike a function),
638 * but its returned value (assumed to be some sort of Scheme expression) is evaluated.
639 *
640 * for s7_make_closure, see the "Closure defined in C" example in s7.html.
641 */

643 /* In s7, (define* (name . args) body) or (define name (lambda* args body))
644 * define a function that takes optional (keyword) named arguments.
645 * The keywords :key and :optional are ok, but they are ignored --
646 * they exist to be compatible with other define* implementations.
647 * The "args" is a list that can contain either names (normal arguments),
648 * or lists of the form (name default-value), in any order. When called,
649 * the names are bound to their default values (or #f), then the function's
650 * current arglist is scanned. Any name that occurs as a keyword (":name")
651 * precedes that argument's new value. Otherwise, as values occur, they
652 * are plugged into the environment based on their position in the arglist
653 * (as normal for a function). So,
654 *
655 * (define* (hi a (b 32) (c "hi")) (list a b c))
656 *
657 * is equivalent to other implementations (define* (hi a :key (b 32) ...))
658 * or (define* (hi a :optional (b 32) ...)) -- these args are all
659 * "optional-key" args in CLM jargon.
660 *
661 * (hi 1) -> '(1 32 "hi")
662 * (hi :b 2 :a 3) -> '(3 2 "hi")
663 * (hi 3 2 1) -> '(3 2 1)
664 *
665 * and so on. :rest causes its argument to be bound to the rest
666 * of the arguments at that point.
667 *
668 * The C connection to this takes the function name, the C function to call, the argument
669 * list as written in Scheme, and the documentation string. s7 makes sure the arguments
670 * are ordered correctly and have the specified defaults before calling the C function.
671 * s7_define_function_star(sc, "a-func", a_func, "arg1 (arg2 32)", "an example of C define*");
672 * Now (a-func :arg1 2) calls the C function a_func(2, 32). See the example program in s7.html.
673 *
674 * In s7 Scheme, define* can be used just for its optional arguments feature, but that is
675 * included in s7_define_function. s7_define_function_star implements keyword arguments
676 * for C-level functions (as well as optional/rest arguments).
677 */
