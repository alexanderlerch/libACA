59Error_t CNmf::compNmf(CNmfResult *pCNmfResult, const float *const *const ppfIn)
60{
61
62 if (!pCNmfResult || !ppfIn)
63 return Error_t::kFunctionInvalidArgsError;
64
65 if (!m_bIsInitialized || !pCNmfResult->isInitialized())
66 return Error_t::kFunctionIllegalCallError;
67
68 enum Cost_t
69 {
70 kStartCost,
71 kPrevCost,
72 kCurrCost,
73
74 kNumCosts
75 };
76 float afCost[kNumCosts] = { 0 };
77
78 for (int k = 0; k < m_iMaxIter; k++)
79 {
80 // this is the heart of it
81 afCost[kCurrCost] = runNmfIter(pCNmfResult, ppfIn);
82
83 assert(!std::isnan(afCost[kCurrCost]));
84
85 // check if we can stop
86 if (k > 0)
87 {
88 if (std::abs((afCost[kCurrCost] - afCost[kPrevCost]) / (afCost[kStartCost] - afCost[kCurrCost] + m_kMinOffset)) < .001F)
89 break;
90 }
91 else
92 afCost[kStartCost] = afCost[kCurrCost];
93
94 afCost[kPrevCost] = afCost[kCurrCost];
95 }
96 return Error_t::kNoError;
97}

99float CNmf::runNmfIter(CNmfResult *pCNmfResult, const float *const *const ppfIn)
100{
101 float **ppfW = pCNmfResult->getMatPointer(CNmfResult::kW);
102 float **ppfH = pCNmfResult->getMatPointer(CNmfResult::kH);
103 float **ppfXHat = pCNmfResult->getMatPointer(CNmfResult::kXHat);
104
105 int iRank = pCNmfResult->getMatCols(CNmfResult::kW);
106 int iNumObs = pCNmfResult->getMatCols(CNmfResult::kH);
107 int iNumBins = pCNmfResult->getMatRows(CNmfResult::kW);
108
109 // there are special cases were we don't want to update both matrices
110 const bool bUpdateH = true,
111 bUpdateW = true;
112
113 // compute current estimate
114 CMatrix::mulMatMat(ppfXHat, ppfW, ppfH, iNumBins, iRank, iRank, iNumObs);
115 CMatrix::addC_I(ppfXHat, m_kMinOffset, iNumBins, iNumObs);
116
117 //m_ppfX
118 CMatrix::copy(m_ppfX, ppfIn, iNumBins, iNumObs);
119 CMatrix::div_I(m_ppfX, ppfXHat, iNumBins, iNumObs);
120
121 // update H
122 if (bUpdateH)
123 {
124 CMatrix::transpose(m_ppfTransp, ppfW, iNumBins, iRank);
125
126 // denominator
127 CMatrix::mulMatMat(m_ppfDenom, m_ppfTransp, m_ppfOnes, iRank, iNumBins, iNumBins, iNumObs);
128 CMatrix::addC_I(m_ppfDenom, m_fSparsity + m_kMinOffset, iRank, iNumObs);
129
130 // numerator
131 CMatrix::mulMatMat(m_ppfNum, m_ppfTransp, m_ppfX, iRank, iNumBins, iNumBins, iNumObs);
132 CMatrix::div_I(m_ppfNum, m_ppfDenom, iRank, iNumObs);
133 CMatrix::mul_I(ppfH, m_ppfNum, iRank, iNumObs);
134
135 // clean small values
136 CMatrix::setZeroBelowThresh(ppfH, iRank, iNumObs, m_kMinOffset);
137 }
138
139 // update W
140 if (bUpdateW)
141 {
142 CMatrix::transpose(m_ppfTransp, ppfH, iRank, iNumObs);
143
144 // denominator
145 CMatrix::mulMatMat(m_ppfDenom, m_ppfOnes, m_ppfTransp, iNumBins, iNumObs, iNumObs, iRank);
146 CMatrix::addC_I(m_ppfDenom, m_kMinOffset, iNumBins, iRank);
147
148 // numerator
149 CMatrix::mulMatMat(m_ppfNum, m_ppfX, m_ppfTransp, iNumBins, iNumObs, iNumObs, iRank);
150 CMatrix::div_I(m_ppfNum, m_ppfDenom, iNumBins, iRank);
151 CMatrix::mul_I(ppfW, m_ppfNum, iNumBins, iRank);
152
153 // normalize
154 CMatrix::vecnorm_I(ppfW, iNumBins, iRank);
155
156 // clean small values
157 CMatrix::setZeroBelowThresh(ppfW, iNumBins, iRank, m_kMinOffset);
158 }
159
160 // compute error
161 CMatrix::mulMatMat(ppfXHat, ppfW, ppfH, iNumBins, iRank, iRank, iNumObs);
162 CMatrix::addC_I(ppfXHat, m_kMinOffset, iNumBins, iNumObs);
163
164 // deal with sparsity
165 CMatrix::addC_I(ppfXHat, m_fSparsity * CMatrix::getNorm(ppfXHat, iNumBins, iNumObs), iNumBins, iNumObs);
166
167 return CMatrix::calcKlDivergence(ppfIn, ppfXHat, iNumBins, iNumObs);
168}
