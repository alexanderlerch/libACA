59Error_t CGmm::compGmm(CGmmResult* pCResult, float** ppfFeatures)
60{
61
62 // init: randomly selected data points as cluster means
63 initState_(ppfFeatures, pCResult);
64
65 // iterate
66 for (auto i = 0; i < m_iMaxIter; i++)
67 {
68 // copy state to prev state
69 PrevState = *pCResult;

84void CGmm::initState_(float** ppfFeatures, CGmmResult* pCCurrState)
85{

109void CGmm::compProbabilities_(float** ppfFeatures, CGmmResult* pCCurrState)
110{
111 // compute gaussian per cluster per observation
112 for (auto k = 0; k < m_iK; k++)
113 {
114 pCCurrState->getSigma(m_appfSigma[0], k);
115 float fNorm = static_cast<float>(1. / std::sqrt(std::pow(2. * M_PI, m_iNumFeatures) * CMatrix::det(m_appfSigma[0], m_iNumFeatures, m_iNumFeatures)));
116 CMatrix::inv_I(m_appfSigma[0], m_iNumFeatures, m_iNumFeatures);
117
118 for (auto n = 0; n < m_iNumObs; n++)
119 {
120 for (auto v = 0; v < m_iNumFeatures; v++)
121 m_apfProc[0][v] = ppfFeatures[v][n] - pCCurrState->getMu(k, v);
122
123 CMatrix::mulMatColVec(m_apfProc[1], m_appfSigma[0], m_apfProc[0], m_iNumFeatures, m_iNumFeatures);
124
125 m_ppfProb[k][n] = fNorm * std::exp(-.5F * CVector::mulScalar(m_apfProc[0], m_apfProc[1], m_iNumFeatures));
126 }
127 CVector::mulC_I(m_ppfProb[k], pCCurrState->getPrior(k), m_iNumObs);
128 }
129
130 // norm over clusters
131 for (auto n = 0; n < m_iNumObs; n++)
132 {
133 float fSum = CMatrix::getSumCol(m_ppfProb, n, m_iK);
134
135 if (fSum > 0)
136 CMatrix::mulColC_I(m_ppfProb, 1.F / fSum, n, m_iK);
137 }
138}

140void CGmm::updateState_(float** ppfFeatures, CGmmResult* pCCurrState)
141{
142 for (auto k = 0; k < m_iK; k++)
143 {
144 // update priors
145 pCCurrState->setPrior(k, CVector::getMean(m_ppfProb[k], m_iNumObs));
146
147 // update means
148 for (auto v = 0; v < m_iNumFeatures; v++)
149 pCCurrState->setMu(k, v, CVector::mulScalar(ppfFeatures[v], m_ppfProb[k], m_iNumObs) / CVector::getSum(m_ppfProb[k], m_iNumObs));
150
151 // update sigma
152 CMatrix::setZero(m_appfSigma[0], m_iNumFeatures, m_iNumFeatures);
153 for (auto n = 0; n < m_iNumObs; n++)
154 {
155 for (auto v = 0; v < m_iNumFeatures; v++)
156 m_apfProc[0][v] = ppfFeatures[v][n] - pCCurrState->getMu(k, v);
157
158 CMatrix::mulColVecRowVec(m_appfSigma[1], m_apfProc[0], m_apfProc[0], m_iNumFeatures, m_iNumFeatures);
159 CMatrix::mulC_I(m_appfSigma[1], m_ppfProb[k][n], m_iNumFeatures, m_iNumFeatures);
160 CMatrix::add_I(m_appfSigma[0], m_appfSigma[1], m_iNumFeatures, m_iNumFeatures);
161 }
162 CMatrix::mulC_I(m_appfSigma[0], 1.F / CVector::getSum(m_ppfProb[k], m_iNumObs), m_iNumFeatures, m_iNumFeatures);
163 pCCurrState->setSigma(k, m_appfSigma[0]);
164 }

167bool CGmm::checkConverged_(CGmmResult* pCCurrState)
168{
169 float fSum = 0;
170 for (auto k = 0; k < m_iK; k++)
171 {
172 for (auto v = 0; v < m_iNumFeatures; v++)
173 fSum += std::abs(pCCurrState->getMu(k, v) - PrevState.getMu(k, v));
174 }
175
176 if (fSum <= 1e-20F)
177 return true;
178
179 return false;
180}
