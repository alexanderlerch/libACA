104 /*! converts a frequency scalar to midi (float)
105 \param fInHz frequency value in Hz
106 \param fA4InHz tuning (reference) frequency in Hz
107 \return Midi value
108 */
109 static float convertFreq2Midi(float fInHz, float fA4InHz = 440.F)
110 {
111 assert(fInHz > 0);
112 assert(fA4InHz > 0);
113 return 69.F + 12.F * std::log2(fInHz / fA4InHz);
114 };

223 /*! converts a float vector in a 32 bit word based on the sign
224 \param pfVec vector of signed float values (length 32)
225 \return uint32_t 32 bit word
226 */
227 static uint32_t convertFloat2Word(const float *pfVec)
228 {
229 assert(pfVec);
230
231 int iLength = 32;
232 uint32_t iRes = 0;
233
234 for (auto b = 0; b < iLength; b++)
235 {
236 if (pfVec[b] > 0)
237 iRes += (1 << (b + 1));
238 }
239
240 return iRes;
241 }

243 /*! converts a 32 bit word into a float vector of -1.F and 1.F
244 \param pfDest destination buffer (length 32, to be written)
245 \param iWord 32 bit word to be decoded
246 */
247 static void convertWord2Float(float *pfDest, uint32_t iWord)
248 {
249 assert(pfDest);
250
251 int iLength = 32;
252
253 for (auto b = 0; b < iLength; b++)
254 {
255 if (iWord & (1 << (b + 1)))
256 pfDest[b] = 1.F;
257 else
258 pfDest[b] = -1.F;
259 }
