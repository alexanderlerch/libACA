146void CGammatone::calcFilterCoeffs_(int iOrder)
147{
148 assert(m_fSampleRate > 0);
149 assert(m_fFreqCenter >= 0);
150
151 const double f2Pi = 2 * M_PI;
152 const double fEarQ = 9.26449;
153 const double fBandwidth = 24.7;
154 const double B = 1.019 * f2Pi * std::pow(std::pow(m_fFreqCenter / fEarQ, iOrder) + std::pow(fBandwidth, iOrder), 1. / iOrder);
155
156 double fArg = f2Pi * m_fFreqCenter / m_fSampleRate;
157 double fCos = std::cos(fArg);
158 double fSin = std::sin(fArg);
159 double fExp = std::exp(B / m_fSampleRate);
160 double fSqrtA = 2 * std::sqrt(3. + std::pow(2., 1.5F));
161 double fSqrtS = 2 * std::sqrt(3. - std::pow(2., 1.5F));
162 double fGain = 0;
163
164 for (auto c = 0; c < kNumFilters; c++)
165 {
166 m_aafCoeffB[c][0] = 1. / m_fSampleRate;
167 m_aafCoeffB[c][2] = 0.;
168
169 m_aafCoeffA[c][0] = 1.;
170 m_aafCoeffA[c][1] = -2. * fCos / fExp;
171 m_aafCoeffA[c][2] = std::exp(-2. * B / m_fSampleRate);
172 }
173
174 m_aafCoeffB[0][1] = -(2. / m_fSampleRate * fCos / fExp + fSqrtA / m_fSampleRate * fSin / fExp) / 2.;
175 m_aafCoeffB[1][1] = -(2. / m_fSampleRate * fCos / fExp - fSqrtA / m_fSampleRate * fSin / fExp) / 2.;
176 m_aafCoeffB[2][1] = -(2. / m_fSampleRate * fCos / fExp + fSqrtS / m_fSampleRate * fSin / fExp) / 2.;
177 m_aafCoeffB[3][1] = -(2. / m_fSampleRate * fCos / fExp - fSqrtS / m_fSampleRate * fSin / fExp) / 2.;
178
179 // apply gain to first cascade
180 {
181 fSqrtA *= .5;
182 fSqrtS *= .5;
183
184 double fScale = 2 * 2 * 2 * 2 / (m_fSampleRate * m_fSampleRate * m_fSampleRate * m_fSampleRate);
185 std::complex<double> num1(-std::cos(2. * fArg), -std::sin(2 * fArg));
186 std::complex<double> num2(std::cos(fArg) * std::exp(-B / m_fSampleRate), std::sin(fArg) * std::exp(-B / m_fSampleRate));
187 std::complex<double> denom(-2. / std::exp(2. * B / m_fSampleRate) + 2. / fExp + std::cos(2 * fArg) * (2. / fExp - 2.), std::sin(2 * fArg) * (2. / fExp - 2.));
188
189 auto numerator = fScale *
190 (num1 + num2 * (fCos - fSqrtS * fSin)) *
191 (num1 + num2 * (fCos + fSqrtS * fSin)) *
192 (num1 + num2 * (fCos - fSqrtA * fSin)) *
193 (num1 + num2 * (fCos + fSqrtA * fSin));
194 auto denominator = denom * denom * denom * denom;
195 fGain = std::abs(numerator / denominator);
196 }
197 for (auto k = 0; k < kNumCoeffs; k++)
198 m_aafCoeffB[0][k] /= fGain;
199
200 for (auto c = 0; c < kNumFilters; c++)
201 m_apCFilter[c]->init(m_aafCoeffB[c], m_aafCoeffA[c], 3);
202}

402Error_t CGammaToneFbIf::process(float** ppfOutput)
403{
404 if (!m_bIsInitialized)
405 return Error_t::kFunctionIllegalCallError;
406 if (!ppfOutput)
407 return Error_t::kFunctionInvalidArgsError;
408 if (!ppfOutput[0])
409 return Error_t::kFunctionInvalidArgsError;
410
411 assert(m_pCBlockAudio);
412 assert(m_pCNormalize);
413
414 auto iNumBlocks = m_pCBlockAudio->getNumBlocks();
415
416 for (auto n = 0; n < iNumBlocks; n++)
417 {
418 // retrieve the next audio block
419 int iNumFrames = m_pCBlockAudio->getNextBlock(m_pfProcessBuff);
420
421 // normalize if specified
422 if (m_pCNormalize)
423 m_pCNormalize->normalizeBlock(m_pfProcessBuff, m_iBlockLength);
424
425 for (auto i = 0; i < m_iNumBands; i++)
426 {
427 assert(ppfOutput[i]);
428 m_ppCGammatone[i]->process(&ppfOutput[i][n*m_iBlockLength], m_pfProcessBuff, iNumFrames);
429 }
430 }
431
432 return Error_t::kNoError;
433}

471inline float CGammaToneFbIf::compMidFreqs_(float fFreqLow, float fFreqHigh, int k) const
472{
473 const float fEarQ = 9.26449F;
474 const float fBandwidth = 24.7F;
475
476 return -(fEarQ * fBandwidth) + std::exp((m_iNumBands-k) * (-std::log(fFreqHigh + fEarQ * fBandwidth) + std::log(fFreqLow + fEarQ * fBandwidth)) / m_iNumBands) * (fFreqHigh + fEarQ * fBandwidth);
